  66.  
  67.                self::$class = $class;
  68.                self::$filename = $class . '.php';
  69.  
  70.              if( in_array( self::$filename, self::$sources ) )
  71.                  return;
  72.  
  73.                array_push( self::$sources, self::$filename );
  74.  
  75.                $comments = array();
  76.                $tokens = token_get_all( self::getSourceCode() );
  77.  
  78.                for( $i=0; $i<count( $tokens ); $i++ ) {
  79.  
  80.                     $token = $tokens[$i];

  82.  
  83.                       list( $code, $value ) = $token;
  84.  
  85.                       switch( $code ) {
  86.  
  87.                               case T_COMMENT:
  88.  
  89.                                      if( preg_match( '/^\\s*#@/', $value ) )
  90.                                            array_push( $comments, $value );
  91.  
  92.                                    break;
  93.  
  94.                               case T_CLASS:
  95.  
  96.                                    if( count( $comments ) ) {
  97.  
  98.                                        self::$classes[$class] = self::parseAnnotations( $comments );
  99.                                        $comments = array();

  101.                                    break;
  102.  
  103.                               case T_VARIABLE:
  104.  
  105.                                    if( count( $comments ) ) {
  106.  
  107.                                           $key = str_replace( '$', '', $token[1] );
  108.                                        self::$properties[$class][ $key ] = self::parseAnnotations( $comments );
  109.                                        $comments = array();
  110.                                    }
  111.                                    break;
  112.  
  113.                               case T_FUNCTION:
  114.  
  115.                                    if( count( $comments ) ) {
  116.  
  117.                                        for( $j=$i; $j<count( $tokens ); $j++ ) {
  118.                                             if( is_array( $tokens[$j] ) ) {
  119.                                                  if( $tokens[$j][0] == T_STRING ) {
  120.                                                       self::$methods[$class][$tokens[$j][1]] = self::parseAnnotations( $comments );
  121.                                                       $comments = array();
  122.                                                       break;
  123.                                                  }
  124.                                             }
  125.                                        }
  126.                                    }
  127.                                     break;
  128.  
  129.                                     /*
  130.                                      * @todo Support annotated interfaces
  131.                             case T_INTERFACE:
  132.  
  133.                                    if( count( $comments ) ) {
  134.  
  135.                                        $this->annotations['interface'] = $this->parseAnnotations( implode( "\n", $comments ) );
  136.                                        $comments = array();
  137.                                    }
  138.                                    break;
  139.                                    */
  140.  
  141.                             case T_DOC_COMMENT;
  142.                             case T_WHITESPACE: 
  143.                             case T_PUBLIC: 
  144.                             case T_PROTECTED: 
  145.                             case T_PRIVATE: 
  146.                             case T_ABSTRACT: 
  147.                             case T_FINAL: 
  148.                             case T_VAR: 
  149.                                 break;
  150.  
  151.                             default:
  152.                                 $comments = array();
  153.                                 break;
  154.                         }
  155.                     }
  156.                     else {
  157.  
  158.                         $comments = array();
  159.                     }
  160.                 }

  260.       /**
  261.        * Parses code extracted from the tokenized PHP file for the presence of
  262.        * AgilePHP annotations.
  263.        * 
  264.        * @param String $text The text/code string to parse
  265.        * @return void 
  266.        * @static
  267.        */
  268.       private static function parseAnnotations( array $items ) {
  269.  
  270.                 $annotations = array();
  271.  
  272.                 foreach( $items as $text ) {
  273.  
  274.                       // Extract the annotation string including the name and property/value declaration
  275.                         preg_match_all( '/^\\s*#@(.*)/', $text, $annotes );
  276.  
  277.                       if( !count( $annotes ) )
  278.                               return;
  279.  
  280.                         foreach( $annotes[1] as $annote ) {
  281.         
  282.                                    // Extract the annotation name
  283.                                    preg_match( '/\w+/', $annote, $className );
  284.         
  285.                                    // Create instance of the annotation class or create a new instance of stdClass
  286.                                    // if the annotation class could not be parsed
  287.                                    $oAnnotation = new $className[0]();
  288.         
  289.                                    // Extract name/value pair portion of the annotation
  290.                                    preg_match_all( '/\((.*=.*\(?\)?)\)/', $annote, $props );
  291.         
  292.                                // Extract arrays
  293.                                if( count( $props ) && count( $props[1] ) )
  294.                                       preg_match_all( '/[_a-zA-Z]+[0-9_]?\\s?=\s?{+?.*?}+\\s?,?/', $props[1][0], $arrays );
  295.  
  296.                                // Extract other annotations
  297.                                // @todo Support child annotations
  298.                                //preg_match_all( '/@(.*)?,?/', $props[1][0], $childAnnotes );
  299.         
  300.                                // Add arrays to annotation instance and remove it from the properties
  301.                                    if( isset( $arrays[0] ) ) {
  302.         
  303.                                           $result = self::parseKeyArrayValuePairs( $oAnnotation, $arrays[0], $props[1][0] );
  304.                                           $oAnnotation = $result->annotation;
  305.                                           $props[1][0] = $result->properties;
  306.                                }
  307.         
  308.                                // Add strings and PHP literals to annotation instance
  309.                                if( count( $props ) && count( $props[1] ) )
  310.                                       $oAnnotation = self::parseKeyValuePairs( $oAnnotation, $props[1][0] );
  311.         
  312.                                // Push the annotation instance onto the stack
  313.                                    array_push( $annotations, $oAnnotation );
  314.                         }
  315.                 }
  316.  
  317.                 return $annotations;
  318.       }

  320.       /**
  321.        * Parses an annotations property assignments which contain one or more array values. The
  322.        * array is added to the annotation instance according to its property name and the array
  323.        * is removed from the properties string.
  324.        * 
  325.        * @param Object $oAnnotation An instance of the annotation object
  326.        * @param String $arrays The string value containing each of the property assignments
  327.        * @param String $properties The annotations properties as they were parsed from the code
  328.        * @return stdClass instance containing the annotation instance and truncated properties string
  329.        * @static
  330.        */
  331.       private static function parseKeyArrayValuePairs( $oAnnotation, $arrays, $properties ) {
  332.  
  333.                foreach( $arrays as $array ) {
  334.  
  335.                        // Remove arrays from the parsed annotation property/value assignments
  336.                     $properties = preg_replace( '/' . $array . '/', '', $properties ) . PHP_EOL;
  337.  
  338.                        // Split the array into key/value
  339.                        //preg_match( '/(.*)={1}\s?\{(.*)\},?/', $array, $matches );
  340.                        preg_match( '/(.*)=\s*?\{(.*)\},?/', $array, $matches );
  341.                        $property = trim( $matches[1] );
  342.                        $elements = explode( ',', trim( $matches[2] ) );
  343.  
  344.                        // Place each of the annotations array elements into a PHP array
  345.                      $value = array();
  346.                        foreach( $elements as $element ) {
  347.     
  348.                                $pos = strpos( $element, '=' );
  349.                                if( $pos !== false ) {
  350.     
  351.                                    // Associative array element
  352.                                    $pieces = explode( '=', $element );
  353.                                    //$value[ trim( $pieces[0] ) ] = trim( $pieces[1] );
  354.                                    $value[ trim( $pieces[0] ) ] = self::getQuotedStringValue( $pieces[1] );
  355.                                }
  356.                                else
  357.  
  358.                                    // Indexed array element
  359.                                    array_push( $value, self::getQuotedStringValue( $element ) );
  360.                        }
  361.  
  362.                        // Set the annotation instance property with the PHP array
  363.                        $oAnnotation->$property = $value;
  364.                }
  365.  
  366.                $stdClass = new stdClass();
  367.                $stdClass->annotation = $oAnnotation;
  368.                $stdClass->properties = $properties;
  369.  
  370.                return $stdClass;
  371.       }
  372.  
  373.       /**
  374.        * Parses quoted strings from annotations property VALUE definitons.
  375.        * 
  376.        * @param String $value The value to parse
  377.        * @return void 
  378.        * @static
  379.        */
  380.       private static function getQuotedStringValue( $value ) {
  381.  
  382.                 // Single quoted value
  383.                   $pos = strpos( $value, '\'' );
  384.                   if( $pos !== false ) {
  385.  
  386.                       preg_match( '/\s?\'+(.*)\'+/', $value, $matches );
  387.                          if( count( $matches ) == 2 )
  388.                              return $matches[1];
  389.                    }
  390.  
  391.                    // Double quoted value
  392.                $pos = strpos( $value, '"' );
  393.                    if( $pos !== false ) {
  394.  
  395.                        preg_match( '/\s?"+(.*)"+/', $value, $matches );
  396.                           if( count( $matches ) == 2 )
  397.                                return $matches[1];
  398.                    }
  399.  
  400.                    // Treat unquoted values as objects
  401.                //$o = str_replace( ' ', '', $value );
  402.                $o = preg_replace( '/new/i', '', $value );
  403.                $o = preg_replace( '/\(\)/i', '', trim( $o ) );
  404.                
  405.                if( $o ) return new $o;
  406.       }
  407.  
  408.       /**
  409.        * Parses strings and PHP literals from annotation property definition(s).
  410.        * 
  411.        * @param Object $oAnnotation An instance of the annotation object
  412.        * @param String $properties String representation of the annotations property definition(s).
  413.        * @return The annotation instance populated according to its definition(s).
  414.        * @static
  415.        */
  416.       private static function parseKeyValuePairs( $oAnnotation, $properties ) {
  417.  
  418.                  $keyValuePairItems = explode( ',', $properties );
  419.  
  420.                  foreach( $keyValuePairItems as $kv ) {
  421.  
  422.                              $pieces = explode( '=', $kv );
  423.  
  424.                              preg_match( '/(.*)=(.*)/', $kv, $pieces );
  425.  
  426.                              if( count( $pieces ) < 2 ) continue;
  427.  
  428.                              $property = trim( $pieces[1] );
  429.                              $value = trim( $pieces[2] );
  430.  
  431.                              // Single quoted value
  432.                              $pos = strpos( $value, '\'' );
  433.                              if( $pos !== false ) {
  434.  
  435.                                  preg_match( '/^\'(.*)\'/', $value, $matches );
  436.                                  if( count( $matches ) == 2 ) {
  437.  
  438.                                      $oAnnotation->$property = $matches[1];
  439.                                      continue;
  440.                                  }
  441.                              }
  442.  
  443.                              // Double quoted value
