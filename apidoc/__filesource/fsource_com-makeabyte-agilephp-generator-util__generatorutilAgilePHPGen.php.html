  78.                 if( !file_exists( $this->cacheFile ) )
  79.                        return false;
  80.  
  81.               $cache = null;
  82.               $h = fopen( $this->cacheFile, 'r' );
  83.               while( !feof( $h ) )
  84.                      $cache .= fgets( $h, 1024 );
  85.               fclose( $h );
  86.  
  87.               return unserialize( $cache );
  88.       }

  134.       protected function fixLineBreaks( $file ) {
  135.           
  136.                   if( substr( getcwd(), 0, 1 ) != '/' ) {
  137.  
  138.                        $h = fopen( $file, 'r' );
  139.                       $data = '';
  140.                       while( !feof( $h ) )
  141.                               $data .= fgets( $h, 4096 );
  142.                       fclose( $h );
  143.  
  144.                       $data = str_replace( "\n", PHP_EOL, $data );
  145.  
  146.                      $h = fopen( $file, 'w' );
  147.                       fwrite( $h, $data );
