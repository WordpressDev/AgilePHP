  23.  * Logs a call state context both before and after a method invocation.
  24.  * 
  25.  * @author Jeremy Hahn
  26.  * @copyright Make A Byte, inc
  27.  * @package com.makeabyte.agilephp.interception.interceptors
  28.  */
  29. #@Interceptor
  30. class Audit {
  31.  
  32.       /**
  33.        * Sets the logging level
  34.        * 
  35.        * @var string $level Optional logging level (info|warn|error|debug). Defaults to 'debug'.
  36.        */
  37.       public $level = 'debug';
  38.  
  39.       /**
  40.        * Flag indicating whether or not to display detailed information about the invocation
  41.        * 
  42.        * @var bool $verbose True to display the entire invocation context, false to display class name,
  43.        *                      method name, and arguments. The return value will also be captured and logged.
  44.        *                       Defaults to false (not verbose).
  45.        */
  46.       public $verbose = false;
  47.  
  48.       private $valid = array( 'info', 'warn', 'error', 'debug' );
  49.  
  50.       #@AroundInvoke
  51.       public function logInvocation( InvocationContext $ic ) {
  52.  
  53.                $level = (in_array( $this->level, $this->valid)) ? $this->level : 'debug';
  54.                if( $this->verbose ) {
  55.  
  56.                     Logger::$level( '#@Audit::logInvocation' );
  57.                     Logger::$level( $ic );
  58.                     return $ic->proceed();     
  59.                }
  60.  
  61.                $callee = $ic->getCallee();
  62.                $message = 'class = \'' . $callee['class'] . '\', method = \'' . $callee['function'] . '\', args = \'' . implode( ',', $ic->getParameters() ) . '\'';
  63.  
  64.                Logger::$level( '#@Audit::logInvocation ' . $message ); 
  65.                return $ic->proceed();
  66.       }
  67.  
  68.       #@AfterInvoke
  69.       public function logReturn( InvocationContext $ic ) {
  70.  
  71.                $level = (in_array( $this->level, $this->valid)) ? $this->level : 'debug';
  72.                if( $this->verbose ) {
  73.  
  74.                     Logger::$level( '#@Audit::logReturn' );
  75.                     Logger::$level( $ic );
  76.                     return $ic->proceed();     
  77.                }
  78.  
  79.                $return = $ic->getReturn();
  80.                Logger::$level( '#@Audit::logReturn' );
  81.                Logger::$level( (($return) ? $return : '(null)') );
  82.  
  83.                return $ic->proceed();
  84.       }
  85. }
