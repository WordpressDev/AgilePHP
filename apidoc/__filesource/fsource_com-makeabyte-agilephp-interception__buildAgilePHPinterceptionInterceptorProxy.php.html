  23.  * Proxy applied to intercepted classes
  24.  * 
  25.  * @author Jeremy Hahn
  26.  * @copyright Make A Byte, inc.

  29. class InterceptorProxy {
  30.  
  31.       private $object;                // Stores the intercepted object
  32.       private static $instance;          // Stores a static instance of the intercepted class
  33.  
  34.       /**
  35.        * Initalizes the intercepted class by creating a new instance, passing in
  36.        * any constructor arguments as required. Class level interceptors are invoked
  37.        * here, as well as dependancy injections via #@In.
  38.        * 
  39.        * @return void 
  40.        */
  41.       public function __construct() {
  42.  
  43.                $proxiedClass = get_class( $this );
  44.                $intercepted = get_class( $this ) . '_Intercepted';
  45.  
  46.                // Create the intercepted class using constructor arguments if we have any
  47.                if( $args = func_get_args() ) {
  48.  
  49.                     $interceptedClass = new \ReflectionClass( $intercepted );
  50.                    $this->object = $interceptedClass->newInstanceArgs( $args );
  51.                }
  52.                else
  53.                    $this->object = new $intercepted();
  54.  
  55.                 $class = new \ReflectionClass( $this->object );
  56.  
  57.                foreach( \AgilePHP::getFramework()->getInterceptions() as $interception ) {
  58.  
  59.                         // Invoke class level interceptors
  60.                       if( $interception->getClass() == $proxiedClass &&
  61.                            !$interception->getMethod() && !$interception->getProperty() ) {
  62.  
  63.                             $interceptorClass = new \AnnotatedClass( $interception->getInterceptor() );
  64.                              foreach( $interceptorClass->getMethods() as $interceptorMethod ) {
  65.  
  66.                                        if( $interceptorMethod->hasAnnotation( 'AroundInvoke' ) ) {
  67.  
  68.                                              $invocationCtx = new \InvocationContext( $this->object, null, null, $interception->getInterceptor() );
  69.                                       $ctx = $interceptorMethod->invoke( $interception->getInterceptor(), $invocationCtx );
  70.                                       if( $ctx instanceof InvocationContext && $ctx->proceed )
  71.                                             $this->object = $ctx->getTarget();
  72.                                       }
  73.                             }
  74.                       }
  75.  
  76.                       // Perform property/field injections
  77.                       if( $interception->getClass() == $proxiedClass && $interception->getProperty() ) {
  78.  
  79.                                 // Execute property level interceptors
  80.                                 $p = new \ReflectionProperty( $this->object, $interception->getProperty() );
  81.                                 if( !$p->isPublic() )
  82.                                      throw new \AgilePHP_InterceptionException( 'Property level interceptor requires public context at \'' . $proxiedClass .
  83.                                               '::' . $interception->getProperty() . '\'.' );
  84.  
  85.                                 $interceptorClass = new \AnnotatedClass( $interception->getInterceptor() );
  86.                                 foreach( $interceptorClass->getMethods() as $interceptorMethod ) {
  87.  
  88.                                           if( $interceptorMethod->hasAnnotation( 'AroundInvoke' ) ) {
  89.  
  90.                                                 $invocationCtx = new \InvocationContext( $this->object, null, null, $interception->getInterceptor(), $interception->getProperty() );
  91.                                          $value = $interceptorMethod->invoke( $interception->getInterceptor(), $invocationCtx );
  92.                                          $p->setValue( $this->object, $value );                                               
  93.                                          }
  94.                                          if( $interceptorMethod->hasAnnotation( 'AfterInvoke' ) ) {
  95.  
  96.                                                 $invocationCtx = new \InvocationContext( $this->object, null, null, $interception->getInterceptor(), $interception->getProperty() );
  97.                                            $interceptorMethod->invoke( $interception->getInterceptor(), $invocationCtx  );
  98.                                            }
  99.                                 }
  100.                       }
  101.              }
  102.       }
  103.  
  104.       /**
  105.        * Returns a singleton instance of the intercepted class
  106.        * 
  107.        * @return Singleton instance of the intercepted class
  108.        * @static
  109.        */
  110.       public static function getInstance() {
  111.  
  112.                $backtrace = debug_backtrace();
  113.                $name = $backtrace[0]['class'] . '_Intercepted';
  114.  
  115.              $class = new \ReflectionClass( $name );
  116.                $m = $class->getMethod( 'getInstance' );
  117.                $args = func_get_args();
  118.                self::$instance = ($args) ? $m->invokeArgs( $class, $args ) : $m->invoke( $class );
  119.  
  120.                return self::$instance;
  121.        }
  122.  
  123.       /**
  124.        * Returns the instance of the intercepted class. This is the original class
  125.        * that was requested, renamed to '<class_name>_Intercepted'.
  126.        * 
  127.        * @return The intercepted class instance
  128.        */
  129.       public function getInterceptedInstance() {
  130.  
  131.                return $this->object;
  132.       }
  133.  
  134.       /**
  135.        * Magic PHP property accessor used to intercept calls to properties.
  136.        * 
  137.        * @param String $property The property/field name being accessed
  138.        * @return The property/field value
  139.        * @throws AgilePHP_InterceptionException
  140.        */
  141.       public function __get( $property ) {
  142.  
  143.                try {
  144.                      $rp = new \ReflectionProperty( $this->object, $property );
  145.                        return $rp->getValue( $this->object );
  146.                }
  147.                catch( ReflectionException $re ) {
  148.  
  149.                        throw new \AgilePHP_InterceptionException( $re->getMessage(), $re->getCode() );
  150.                }
  151.         }
  152.  
  153.         /**
  154.          * Magic PHP property mutator used to intercept mutation calls.
  155.          * 
  156.          * @param String $property The property/field name being set
  157.          * @param mixed $value The value to set
  158.          * @return void 
  159.          * @throws AgilePHP_InterceptionException
  160.          */
  161.         public function __set( $property, $value ) {
  162.  
  163.                  try {
  164.                      $rp = new \ReflectionProperty( $this->object, $property );
  165.                        return $rp->setValue( $this->object, $value );
  166.                }
  167.                catch( ReflectionException $re ) {
  168.  
  169.                        throw new \AgilePHP_InterceptionException( $re->getMessage(), $re->getCode() );
  170.                }
  171.       }
  172.  
  173.       /**
  174.        * Magic PHP isset. Checks a property to see if its set
  175.        * 
  176.        * @param $property The property/field being tested
  177.        * @return True if the property/field is set, false otherwise
  178.        * @throws AgilePHP_InterceptionException
  179.        */
  180.         public function __isset( $property ) {
  181.  
  182.                  return $this->__get( $property ) ? true : false;
  183.         }
  184.  
  185.         /**
  186.          * Magic PHP unset function used to intercept unset calls.
  187.          * 
  188.          * @param String $property The property/field being unset
  189.          * @return void 
  190.          * @throws AgilePHP_InterceptionException
  191.          */
  192.         public function __unset( $property ) {
  193.  
  194.                 $this->__set( $property, null );
  195.         }
  196.  
  197.         /**
  198.          * Magic PHP method executor used to intercept method calls.
  199.          * 
  200.          * @param String $method The method being called
  201.          * @param Array $args The arguments being passed
  202.          * @return The result of the intercepted method invocation
  203.          * @throws AgilePHP_InterceptionException
  204.          */
  205.       public function __call( $method, $args ) {
  206.  
  207.                $class = new \ReflectionClass( $this->object );
  208.  
  209.                // Invoke interceptor if AgilePHP contains an Interception for this method call
  210.                $interceptions = \AgilePHP::getFramework()->getInterceptions();
  211.              foreach( $interceptions as $interception ) {
  212.  
  213.                        if( $interception->getClass() == get_class( $this ) && $interception->getMethod() == $method ) {
  214.  
  215.                                  $interceptorClass = new \AnnotatedClass( $interception->getInterceptor() );
  216.                                  foreach( $interceptorClass->getMethods() as $interceptorMethod ) {
  217.  
  218.                                             $invoked = false;
  219.                                             $invocationCtx = new \InvocationContext( $this->object, $method, $args, $interception->getInterceptor() );
  220.  
  221.                                             if( $interceptorMethod->hasAnnotation( 'AroundInvoke' ) ) {
  222.  
  223.                                            $ctx = $interceptorMethod->invoke( $interception->getInterceptor(), $invocationCtx );
  224.                                            $invocationCtx = $ctx;
  225.  
  226.                                            // Only execute the intercepted __call if the InvocationContext has had its proceed() method invoked.
  227.                                            if( $ctx instanceof \InvocationContext && $ctx->proceed ) {
  228.  
  229.                                                $m = $class->getMethod( $ctx->getMethod() );
  230.                                                if( $m !== null ) {
  231.  
  232.                                                       // Capture the return value and update the InvocationContext with the return value for #@AroundInvoke
  233.                                                      $return = $args ? $m->invokeArgs( $this->object, $ctx->getParameters() ) : $m->invoke( $this->object );
  234.                                                      $invocationCtx->setReturn( $return );
  235.                                                      $invoked = true;
  236.                                                }
  237.                                            }
  238.                                            }
  239.  
  240.                                            // Execute #@AfterInvoke interceptor methods passing in the InvocationContext as it
  241.                                            // was returned from the #@AroundInvoke operation if it exists. Otherwise the initial
  242.                                            // call to #@AroundInvoke is passed into #@AfterInvoke.
  243.                                            foreach( $interceptorClass->getMethods() as $interceptorMethod ) {
  244.  
  245.                                               if( $interceptorMethod->hasAnnotation( 'AfterInvoke' ) ) {
  246.  
  247.                                                $ctx = $interceptorMethod->invoke( $interception->getInterceptor(), $invocationCtx );
  248.                                                if( $ctx instanceof InvocationContext && $ctx->proceed )
  249.                                                           return $ctx->getReturn();
  250.                                            }
  251.                                            }
  252.  
  253.                                            if( $invoked ) return $return;
  254.                                  }
  255.                             }
  256.              }
  257.  
  258.                // No interceptors, invoke the intercepted method as it was called.
  259.              $m = $class->getMethod( $method );
  260.              return $args ? $m->invokeArgs( $this->object, $args ) : $m->invoke( $this->object, $args );
  261.       }
  262.  
  263.       /**
  264.        * Invokes class level interceptor #@AfterInvoke methods upon destruction.
  265.        * 
  266.        * @return void 
  267.        */
  268.       public function __destruct() {
  269.  
  270.                $proxiedClass = get_class( $this );
  271.                foreach( \AgilePHP::getFramework()->getInterceptions() as $interception ) {
  272.  
  273.                          // Invoke class level interceptor #@AfterInvoke
  274.                        if( $interception->getClass() == $proxiedClass &&
  275.                                 !$interception->getMethod() && !$interception->getProperty() ) {
  276.  
  277.                              $interceptorClass = new \AnnotatedClass( $interception->getInterceptor() );
  278.                               foreach( $interceptorClass->getMethods() as $interceptorMethod ) {
  279.  
  280.                                        if( $interceptorMethod->hasAnnotation( 'AfterInvoke' ) ) {
  281.  
  282.                                               $invocationCtx = new \InvocationContext( $this->object, null, null, $interception->getInterceptor() );
  283.                                          $interceptorMethod->invoke( $interception->getInterceptor(), $invocationCtx  );
  284.                                          }
  285.                              }
  286.                        }
  287.              }
  288.       }
