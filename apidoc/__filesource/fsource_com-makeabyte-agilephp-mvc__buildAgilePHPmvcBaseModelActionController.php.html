  238.          /**
  239.           * Parses $ids sent in by the action method and sets all primary keys
  240.           * in the model defined in the extension class.
  241.           * 
  242.           * @param String $ids The 'ids' as passed in from the form submit. These 'ids' are
  243.           *                     automagically generated by BaseModelXslController.
  244.           * @return void 
  245.           */
  246.          protected function setPrimaryKeys( $ids ) {
  247.  
  248.                      if( $ids == null ) return;
  249.  
  250.                      $table = $this->getPersistenceManager()->getTableByModel( $this->getModel() );
  251.                      $pkeyColumns = $table->getPrimaryKeyColumns();
  252.                      
  253.                      if( !$pkeyColumns ) return;  // This is bad SQL programming but causes an error here when no primary key is set!
  254.  
  255.                      // Single primary key
  256.                      if( strpos( $ids, '_' ) === false ) {
  257.  
  258.                          $mutator = $this->toMutator( $pkeyColumns[0]->getModelPropertyName() );
  259.                           $this->getModel()->$mutator( $ids );
  260.                           return;
  261.                      }
  262.  
  263.                      // Compound primary key
  264.                    $idz = explode( '_', $ids );
  265.                      for( $i=0; $i<count( $idz ); $i++ ) {
  266.  
  267.                             $mutator = $this->toMutator( $pkeyColumns[$i]->getModelPropertyName() );
  268.                             $this->getModel()->$mutator( $idz[$i] );
  269.                      }
  270.          }

  272.          /**
  273.           * Uses the AgilePHP RequestScope component to retrieve POST parameters
  274.           * from a form submit and set the model properties defined in the extension
  275.           * class. This method expects the form element names to match the name
  276.           * of the model's property.
  277.           * 
  278.           * @return void 
  279.           */
  280.          protected function setModelValues() {
  281.  
  282.                      $request = Scope::getInstance()->getRequestScope();
  283.                     $table = $this->getPersistenceManager()->getTableByModel( $this->getModel() );
  284.  
  285.                      if( !$request->getParameters() )
  286.                            return;
  287.  
  288.                         foreach( $table->getColumns() as $column ) {
  289.  
  290.                                     if( !$request->get( $column->getModelPropertyName() ) ) {
  291.  
  292.                                         $mutator = $this->toMutator( $column->getModelPropertyName() );
  293.                                      $this->getModel()->$mutator( null );
  294.                                      continue;
  295.                                     }
  296.  
  297.                                  foreach( $request->getParameters() as $name => $value ) {
  298.         
  299.                                        if( $name == 'AGILEPHP_REQUEST_TOKEN' ) continue;
  300.         
  301.                                        if( $name == 'password1' || $name == 'password2' ) {
  302.         
  303.                                              if( $name == 'password2' ) continue;
  304.         
  305.                                              if( $request->getSanitized( 'password1' ) !== $request->getSanitized( 'password2' ) ) {
  306.         
  307.                                                   $this->getRenderer()->set( 'error', 'Passwords don\'t match' );
  308.                                                    $this->getRenderer()->render( 'error' );
  309.                                                    exit;
  310.                                              }
  311.         
  312.                                              if( $request->getSanitized( 'oldPassword' ) != $value )
  313.                                             $this->getModel()->setPassword( $value );
  314.         
  315.                                         continue;
  316.                                        }
  317.  
  318.                                        $accessor = $this->toAccessor( $name );
  319.                                        $mutator = $this->toMutator( $name );
  320.  
  321.                                          if( $column->getModelPropertyName() == $name ) {
  322.         
  323.                                              $value = ($column->getSanitize() === true) ? 
  324.                                                       urldecode( stripslashes( stripslashes( $request->sanitize( $value ) ) ) ) :
  325.                                                       urldecode( stripslashes( stripslashes( $value ) ) );
  326.  
  327.                                             if( $column->isForeignKey() ) {
  328.  
  329.                                                  $fmodelName = $column->getForeignKey()->getReferencedTableInstance()->getModel();
  330.                                                  $fModel = new $fmodelName();
  331.  
  332.                                                  $casted = $this->getCastedValue( $column->getType(), $value, $column->getSanitize() );
  333.                                               $this->getModel()->$mutator( $casted );
  334.  
  335.                                                  // php namespace support
  336.                                                 $namespace = explode( '\\', $fmodelName );
  337.                                                  $className = array_pop( $namespace );
  338.  
  339.                                             $instanceMutator = $this->toMutator( $className );
  340.                                             $accessor = $this->toMutator( $column->getForeignKey()->getReferencedColumnInstance()->getModelPropertyName() );
  341.                                             $fModel->$accessor( $casted );
  342.                                               $this->getModel()->$instanceMutator( (($casted) ? $fModel : NULL) );
  343.  
  344.                                              continue;
  345.                                          }
  346.  
  347.                                             $this->getModel()->$mutator( $this->getCastedValue( $column->getType(), $value, $column->getSanitize() ) );
  348.                                         }
  349.                                  }
  350.                       }
  351.                       //print_r( $this->getModel() );
  352.          }

  354.          /**
  355.           * Responsible for returning proper data type mappings between SQL and PHP.
  356.           * 
  357.           * @param String $type The column type specified in persistence.xml for the property being set.
  358.           * @param $mutator The mutator method name to invoke on the current model.
  359.           * @param $value The SQL data type being set.
  360.           * @return void 
  361.           */
  362.          protected function getCastedValue( $type, $value, $isSanitized ) {
  363.  
  364.                   if( $value == 'NULL' ) return NULL;
  365.              
  366.                  switch( $type ) {
  367.  
  368.                                case 'boolean':
  369.                                       return $value == true ? true : false;
  370.                                  break;
  371.  
  372.                                case 'integer':
  373.                                case 'tinyiny':
  374.                                case 'smallint':
  375.                                      return (integer) $value;
  376.                                break;
  377.     
  378.                                case 'int':
  379.                                       return (int) $value;
  380.                                break;
  381.  
  382.                                case 'bigint':
  383.                               return $this->getPersistenceManager()->toBigInt( $value );
  384.                          break;
  385.  
  386.                                case 'double':
  387.                                       return (float) $value;
  388.                                break;
  389.  
  390.                                case 'decimal':
  391.                                       return (float) $value;
  392.                                break;
  393.  
  394.                                case 'float':
  395.                                case 'money':
  396.                                       return (float) $value;
  397.                                break;
  398.  
  399.                                case 'text':
  400.                                case 'char':
  401.                                case 'varchar':
  402.                                case 'nvarchar':
  403.                                case 'ntext':
  404.                                       return (string) $value;
  405.                                break;
  406.  
  407.                                case 'date':
  408.                                       return date( 'c', strtotime( $value ) );
  409.                                break;
  410.  
  411.                                case 'datetime':
  412.                                case 'smalldatetime':
  413.                                       return date( 'c', strtotime( $value ) );
  414.                                break;
  415.  
  416.                                case 'bit':
  417.                                       return ($value == 1) ? 1 : 'NULL';
