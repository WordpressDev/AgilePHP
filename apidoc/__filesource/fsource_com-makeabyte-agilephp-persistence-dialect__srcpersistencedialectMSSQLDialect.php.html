  134.       private function toCreateTableSQL( Table $table ) {
  135.  
  136.                 $lengthables = array( 'binary', 'char', 'decimal', 'nchar', 'numeric', 'nvarchar', 'varbinary', 'varchar' );
  137.                 $sql = 'CREATE TABLE ' . $table->getName() . ' ( ';
  138.  
  139.                  foreach( $table->getColumns() as $column ) {
  140.  
  141.                           $sql .= '[' . $column->getName() . '] ' . $column->getType() . 
  142.                                   ((in_array( $column->getType(), $lengthables ) && $column->getLength()) ? '(' . $column->getLength() . ')' : '') .
  143.                                   (($column->isRequired() == true) ? ' NOT NULL' : '') .
  144.                                   (($column->isAutoIncrement() === true) ? ' IDENTITY(1,1)' : '') .
  145.                                   (($column->getDefault() && $column->getType() != 'datetime' && 
  146.                                                $column->getType() != 'timestamp' && !$column->isAutoIncrement() &&
  147.                                                !$column->isPrimaryKey()) ?
  148.                                                ' DEFAULT \'' . $column->getDefault() . '\'' : '') .
  149.                                   ((!$column->getDefault() && !$column->isRequired() && !$column->isAutoIncrement() &&
  150.                                               !$column->isPrimaryKey()) ? ' DEFAULT NULL' : '') . ', ';
  151.                  }
  152.  
  153.                  $pkeyColumns = $table->getPrimaryKeyColumns();
  154.                  if( count( $pkeyColumns ) ) {
  155.  
  156.                        $sql .= ' PRIMARY KEY ( ';
  157.                      for( $i=0; $i<count( $pkeyColumns ); $i++ ) {
  158.  
  159.                           $sql .= '[' . $pkeyColumns[$i]->getName() . ']';
  160.  
  161.                           if( ($i+1) < count( $pkeyColumns ) )
  162.                               $sql .= ', ';
  163.                      }
  164.                      $sql .= ' ), ';
  165.                  }
  166.  
  167.                  if( $table->hasForeignKey() ) {
  168.  
  169.                     $bProcessedKeys = array();
  170.                        $foreignKeyColumns = $table->getForeignKeyColumns();
  171.                        for( $h=0; $h<count( $foreignKeyColumns ); $h++ ) {
  172.  
  173.                               $fk = $foreignKeyColumns[$h]->getForeignKey();
  174.  
  175.                               if( in_array( $fk->getName(), $bProcessedKeys ) )
  176.                                   continue;
  177.  
  178.                               $fk->setOnUpdate( str_replace( '_', ' ', $fk->getOnUpdate() ) );
  179.                               $fk->setOnDelete( str_replace( '_', ' ', $fk->getOnDelete() ) );
  180.  
  181.                                  // Get foreign keys which are part of the same relationship
  182.                                  $relatedKeys = $table->getForeignKeyColumnsByKey( $fk->getName() );
  183.  
  184.                                  $sql .= ' CONSTRAINT ' . $fk->getName() . '';
  185.                              $sql .= ' FOREIGN KEY ( ';
  186.                               for( $j=0; $j<count( $relatedKeys ); $j++ ) {
  187.  
  188.                                         $sql .= '' . $relatedKeys[$j]->getColumnInstance()->getName() . '';
  189.                                       if( ($j+1) < count( $relatedKeys ) )
  190.                                            $sql .= ', ';
  191.                                       }
  192.                             $sql .= ' ) REFERENCES ' . $fk->getReferencedTable() . ' ( ';
  193.                                    for( $j=0; $j<count( $relatedKeys ); $j++ ) {
  194.  
  195.                                          $sql .= '' . $relatedKeys[$j]->getReferencedColumn() . '';
  196.                                               if( ($j+1) < count( $relatedKeys ) )
  197.                                                   $sql .= ', ';
  198.                                    }
  199.                                       $sql .= ' ) ';
  200.                                       $sql .= (($fk->getOnUpdate()) ? ' ON UPDATE ' . $fk->getOnUpdate() : '' );
  201.                                   $sql .= (($fk->getOnDelete()) ? ' ON DELETE ' . $fk->getOnDelete() : '' ) . ', ';
  202.  
  203.                                    array_push( $bProcessedKeys, $fk->getName() );
  204.                         }
  205.                   }
  206.  
  207.                $sql .= ');';
  208.  
  209.                return $sql;
  210.       }

  240.       /**
  241.        * Overrides parent find method to provide MSSQL specific TOP command to limit returned result sets.
  242.        * 
  243.        * @param $model A domain model object. Any fields which are set in the object are used to filter results.
  244.        * @throws AgilePHP_PersistenceException If any primary keys contain null values or any
  245.        *             errors are encountered executing queries
  246.        */
  247.       public function find( $model ) {
  248.  
  249.              $table = $this->getTableByModel( $model );
  250.              $newModel = $table->getModelInstance();
  251.              $values = array();            
  252.  
  253.              Log::debug( 'MSSQLDialect::find Performing find on model \'' . $table->getModel() . '\'.' );
  254.  
  255.                try {
  256.                         $pkeyColumns = $table->getPrimaryKeyColumns();
  257.                          if( $this->isEmpty( $model ) ) {
  258.  
  259.                            $sql = 'SELECT' . ($this->getMaxResults() ? ' TOP ' . $this->getMaxResults() : '') . 
  260.                                               (($this->getDistinct() == null) ? ' *' : 'DISTINCT ' . $this->getDistinct()) . 
  261.                                    ' FROM ' . $table->getName();
  262.  
  263.                            $order = $this->getOrderBy();
  264.                            $offset = $this->getOffset();
  265.                            $groupBy = $this->getGroupBy();
  266.  
  267.                             $sql .= ($this->getRestrictions() != null) ? $this->createRestrictSQL() : '';
  268.                          $sql .= ($order != null) ? ' ORDER BY ' . $order['column'] . ' ' . $order['direction'] : '';
  269.                          $sql .= ($groupBy)? ' GROUP BY ' . $this->getGroupBy() : '';
  270.                             $sql .= ';';
  271.  
  272.                                $this->setDistinct( null );
  273.                             $this->setRestrictions( array() );
  274.                             $this->setRestrictionsLogicOperator( 'AND' );
  275.                             $this->setOrderBy( null, 'ASC' );
  276.                             $this->setGroupBy( null );
  277.                         }
  278.                         else {
  279.                                 $where = '';
  280.  
  281.                                 $columns = $table->getColumns();
  282.                             for( $i=0; $i<count( $columns ); $i++ ) {
  283.  
  284.                                   $accessor = $this->toAccessor( $columns[$i]->getModelPropertyName() );
  285.                                   if( $model->$accessor() == null ) continue;
  286.  
  287.                                   $where .= (count($values) ? ' AND ' : ' ') . $columns[$i]->getName() . '=?';
  288.                                  array_push( $values, $model->$accessor() );
  289.                             }
  290.                             $sql = 'SELECT * FROM ' . $table->getName() . ' WHERE' . $where;
  291.                         }
  292.  
  293.                         // Execute query
  294.                      $this->prepare( $sql );
  295.                      $this->PDOStatement->setFetchMode( PDO::FETCH_OBJ );
  296.                      $result = $this->execute( $values );
  297.  
  298.                      if( !count( $result ) ) {
  299.  
  300.                           Log::debug( 'MSSQLDialect::find Empty result set for model \'' . $table->getModel() . '\'.' );
  301.                           return array();
  302.                      }
  303.  
  304.                       $index = 0;
  305.                       $models = array();
  306.                      foreach( $result as $stdClass  ) {
  307.  
  308.                                $m = $table->getModelInstance();
  309.                                   foreach( get_object_vars( $stdClass ) as $name => $value ) {
  310.  
  311.                                              if( !$value ) continue;
  312.                                              $modelProperty = $this->getPropertyNameForColumn( $table, $name );
  313.  
  314.                                            // Create foreign model instances from foreign values
  315.                                          foreach( $table->getColumns() as $column ) {
  316.  
  317.                                                     if( $column->isForeignKey() && $column->getName() == $name ) {
  318.  
  319.                                                            $foreignModel = $column->getForeignKey()->getReferencedTableInstance()->getModel();
  320.                                                            $foreignInstance = new $foreignModel();
  321.  
  322.                                                            $foreignMutator = $this->toMutator( $column->getForeignKey()->getReferencedColumnInstance()->getModelPropertyName() );
  323.                                                            $foreignInstance->$foreignMutator( $value );
  324.  
  325.                                                            $persisted = $this->find( $foreignInstance );
  326.  
  327.                                                            $instanceMutator = $this->toMutator( $foreignModel );
  328.                                                            $m->$instanceMutator( $persisted[0] );
  329.                                                     }
  330.                                                     else {
  331.  
  332.                                                            $mutator = $this->toMutator( $modelProperty );
  333.                                                              $m->$mutator( $value );
  334.                                                     }
  335.                                          }
  336.                                   }
  337.  
  338.                                   array_push( $models, $m );
  339.                                   $index++;
  340.                                   if( $index == $this->getMaxResults() )  break;
  341.                      }
  342.  
  343.                      return $models;
  344.                }
  345.                catch( Exception $e ) {
  346.  
  347.                        throw new AgilePHP_PersistenceException( $e->getMessage(), $e->getCode() );
  348.                }
  349.  
  350.                return null;
  351.       }

  353.       public function reverseEngineer() {

  355.                $pm = new PersistenceManager(); // execute sproc in different context

  366.                $stmt = $this->query( 'select * from information_schema.tables;' );

  368.              $tables = $stmt->fetchAll();
  369.  
  370.              $stmt2 = $this->query( 'SELECT 
  371.                                 FK_Table  = FK.TABLE_NAME,

  373.                                 PK_Table  = PK.TABLE_NAME, 
  374.                                 PK_Column = PT.COLUMN_NAME,

  384.                                 INNER JOIN 
  385.                                 INFORMATION_SCHEMA.KEY_COLUMN_USAGE CU 
  386.                                     ON C.CONSTRAINT_NAME = CU.CONSTRAINT_NAME 
  387.                                 INNER JOIN 
  388.                                 ( 
  389.                                     SELECT 
  390.                                         i1.TABLE_NAME, i2.COLUMN_NAME 
  391.                                     FROM 
  392.                                         INFORMATION_SCHEMA.TABLE_CONSTRAINTS i1 
  393.                                         INNER JOIN 
  394.                                         INFORMATION_SCHEMA.KEY_COLUMN_USAGE i2
