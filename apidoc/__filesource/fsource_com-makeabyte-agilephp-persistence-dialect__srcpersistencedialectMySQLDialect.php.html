  39.       public function __construct( Database $db ) {
  40.  
  41.                try {
  42.                         $this->pdo = new PDO( 'mysql:host=' . $db->getHostname() . ';dbname=' . $db->getName(), $db->getUsername(), $db->getPassword() );
  43.                         $this->connectFlag = 1;           
  44.                }
  45.                catch( PDOException $pdoe ) {
  46.  
  47.                        Log::debug( 'MySQLDialect::__construct Warning about \'' . $pdoe->getMessage() . '\'.' );
  48.  
  49.                        // If the database doesnt exist, try a generic connection to the server. This allows the create() method to
  50.                        // be invoked to create the database schema.

  52.  
  53.                            $this->pdo = new PDO( 'mysql:host=' . $db->getHostname() . ';', $db->getUsername(), $db->getPassword() );

  57.  
  58.                            $this->connectFlag = -1;
  59.                            throw new AgilePHP_Exception( 'Failed to create MySQLDialect instance. ' . $pdoe->getMessage() );

  86.

  88.                $this->pdo = new PDO( 'mysql:host=' . $this->database->getHostname() . ';dbname=' . $this->database->getName(),
  89.                                         $this->database->getUsername(), $this->database->getPassword() );
  90.  
  91.              $this->query( 'SET foreign_key_checks = 0;' );
  92.  
  93.                foreach( $this->database->getTables() as $table )
  94.                          $this->query( $this->toCreateTableSQL( $table ) );

  109.       /**
  110.        * Generates SQL CREATE TABLE for the specified table.
  111.        * 
  112.        * @param Table $table The table to generate the CREATE TABLE code for
  113.        * @return string The SQL CREATE TABLE code
  114.        */
  115.       private function toCreateTableSQL( Table $table ) {
  116.  
  117.                 $defaultKeywords = array( 'CURRENT_TIMESTAMP' );  // Default values that get passed unquoted
  118.  
  119.                  $sql = 'CREATE TABLE `' . $table->getName() . '` ( ';
  120.  
  121.                  foreach( $table->getColumns() as $column ) {
  122.  
  123.                           $sql .= '`' . $column->getName() . '` ' . $column->getType() . 
  124.                                   (($column->getLength()) ? '(' . $column->getLength() . ')' : '') .
  125.                                   (($column->isRequired() == true) ? ' NOT NULL' : '') .
  126.                                   (($column->isAutoIncrement() === true) ? ' AUTO_INCREMENT' : '') .
  127.                                   (($column->getDefault()) ? ' DEFAULT ' . (in_array($column->getDefault(),$defaultKeywords) ? $column->getDefault() : '\'' . $column->getDefault() . '\'') . '': '') .
  128.                                   ((!$column->getDefault() && !$column->isRequired()) ? ' DEFAULT NULL' : '') . ', ';
  129.                  }
  130.  
  131.                  $pkeyColumns = $table->getPrimaryKeyColumns();
  132.                  if( count( $pkeyColumns ) ) {
  133.  
  134.                        $sql .= ' PRIMARY KEY ( ';
  135.                      for( $i=0; $i<count( $pkeyColumns ); $i++ ) {
  136.  
  137.                           $sql .= '`' . $pkeyColumns[$i]->getName() . '`';
  138.  
  139.                               if( ($i+1) < count( $pkeyColumns ) )
  140.                                      $sql .= ', ';
  141.                      }
  142.                      $sql .= ' )';
  143.  
  144.                      /*
  145.                      if( count( $pkeyColumns ) > 1 )
  146.                            $sql .= ', UNIQUE KEY `' . $pkeyColumns[0]->getName() . '` (`' . $pkeyColumns[0]->getName() . '`)';
  147.                      */
  148.                  }
  149.  
  150.                  if( $table->hasForeignKey() ) {
  151.  
  152.                     $bProcessedKeys = array();
  153.                        $foreignKeyColumns = $table->getForeignKeyColumns();
  154.                        for( $h=0; $h<count( $foreignKeyColumns ); $h++ ) {
  155.  
  156.                                   $fk = $foreignKeyColumns[$h]->getForeignKey();
  157.  
  158.                                   if( in_array( $fk->getName(), $bProcessedKeys ) )
  159.                                       continue;
  160.  
  161.                                      // Get foreign keys which are part of the same relationship
  162.                                      $relatedKeys = $table->getForeignKeyColumnsByKey( $fk->getName() );
  163.  
  164.                                      $sql .= ', KEY `' . $fk->getName() . '` ( ';
  165.  
  166.                                      for( $j=0; $j<count( $relatedKeys ); $j++ ) {
  167.  
  168.                                              array_push( $bProcessedKeys, $relatedKeys[$j]->getName() );
  169.                                              $sql .= '`' . $relatedKeys[$j]->getColumnInstance()->getName() . '`';
  170.                                              if( ($j+1) < count( $relatedKeys ) )
  171.                                                  $sql .= ', ';
  172.                                      }
  173.                                      $sql .= ' ), CONSTRAINT `' . $fk->getName() . '`';
  174.                                          $sql .= ' FOREIGN KEY ( ';
  175.                                   for( $j=0; $j<count( $relatedKeys ); $j++ ) {
  176.  
  177.                                               $sql .= '`' . $relatedKeys[$j]->getColumnInstance()->getName() . '`';
  178.                                              if( ($j+1) < count( $relatedKeys ) )
  179.                                                  $sql .= ', ';
  180.                                      }
  181.                            $sql .= ' ) REFERENCES `' . $fk->getReferencedTable() . '` ( ';
  182.                                   for( $j=0; $j<count( $relatedKeys ); $j++ ) {
  183.  
  184.                                               $sql .= '`' . $relatedKeys[$j]->getReferencedColumn() . '`';
  185.                                              if( ($j+1) < count( $relatedKeys ) )
  186.                                                   $sql .= ', ';
  187.                                   }
  188.                                      $sql .= ' ) ';
  189.                                      $sql .= (($fk->getOnUpdate()) ? ' ON UPDATE ' . $fk->getOnUpdate() : '' );
  190.                                      $sql .= (($fk->getOnDelete()) ? ' ON DELETE ' . $fk->getOnDelete() : '' );
  191.  
  192.                                   array_push( $bProcessedKeys, $fk->getName() );
  193.                        }
  194.                  }
  195.  
  196.                  $engineType = ($table->hasForeignKey() || $table->hasForeignKeyReferences()) ? 'INNODB' : 'MYISAM';
  197.               $sql .= ') ENGINE=' . $engineType . ' DEFAULT CHARSET=latin1;';
  198.  
  199.                  return $sql;
  200.       }

  222.        * @see src/persistence/dialect/SQLDialect#reverseEngineer()
  223.        */
  224.       public function reverseEngineer() {
  225.  
  226.                $Database = new Database();
  227.                $Database->setId( $this->database->getId() );
  228.                $Database->setName( $this->database->getName() );
  229.                $Database->setType( $this->database->getType() );
  230.                $Database->setHostname( $this->database->getHostname() );
  231.                $Database->setUsername( $this->database->getUsername() );
  232.                $Database->setPassword( $this->database->getPassword() );
  233.  
  234.                $stmt = $this->prepare( 'SHOW TABLES' );
  235.                $stmt->execute();
  236.                $stmt->setFetchMode( PDO::FETCH_OBJ );
  237.                $tables = $stmt->fetchAll();
  238.  
  239.                $tblIndex = 'Tables_in_' . $this->getDatabase()->getName();
  240.  
  241.                $fkeyQuery  = 'SELECT
  242.                                 kcu.CONSTRAINT_NAME AS \'constraint\',
  243.                                 kcu.TABLE_NAME AS \'table\',
  244.                                 kcu.COLUMN_NAME AS \'column\',
  245.                                 kcu.REFERENCED_TABLE_NAME AS \'referenced_table\',
  246.                                 kcu.REFERENCED_COLUMN_NAME AS \'referenced_column\',
  247.                                 rc.UPDATE_RULE AS \'update_rule\',
  248.                                 rc.DELETE_RULE AS \'delete_rule\'
  249.                             FROM information_schema.key_column_usage AS kcu
  250.                             INNER JOIN information_schema.REFERENTIAL_CONSTRAINTS AS rc
  251.                             WHERE kcu.TABLE_SCHEMA = rc.CONSTRAINT_SCHEMA
  252.                               AND kcu.CONSTRAINT_NAME = rc.CONSTRAINT_NAME
  253.                               AND rc.CONSTRAINT_SCHEMA = \'' . $Database->getName() . '\'
  254.                               AND kcu.REFERENCED_TABLE_NAME IS NOT NULL';
  255.  
  256.                $stmt = $this->query( $fkeyQuery );
  257.                $stmt->setFetchMode( PDO::FETCH_OBJ );
  258.              $foreignKeys = $stmt->fetchAll();
  259.          
  260.                foreach( $tables as $sqlTable ) {
  261.  
  262.                          $Table = new Table();
  263.                          $Table->setName( str_replace( ' ', '_', $sqlTable->$tblIndex ) );
  264.                          $Table->setModel( ucfirst( $Table->getName() ) );
  265.  
  266.                           $stmt = $this->query( 'DESC ' . $sqlTable->$tblIndex );
  267.                           $stmt->setFetchMode( PDO::FETCH_OBJ );
  268.                           $descriptions = $stmt->fetchAll();
  269.                            
  270.                           foreach( $descriptions as $desc ) {
  271.  
  272.                                   $type = $desc->Type;
  273.                                $length = null;
  274.                                $pos = strpos( $desc->Type, '(' );
  275.     
  276.                                if( $pos !== false ) {
  277.                                     
  278.                                       $type = preg_match_all( '/^(.*)\((.*)\)$/i', $desc->Type, $matches );
  279.                                       
  280.                                       $type = $matches[1][0];
  281.                                       $length = $matches[2][0];
  282.                                }
  283.  
  284.                                $Column = new Column( null, $Table->getName() );
  285.                            $Column->setName( $desc->Field );
  286.                            $Column->setType( $type );
  287.                            $Column->setLength( $length );
  288.  
  289.                            if( isset( $desc->Default ) && $desc->Default )
  290.                                    $Column->setDefault( $desc->Default );
  291.  
  292.                            if( isset( $desc->Null ) && $desc->Null == 'NO' )
  293.                                   $Column->setRequired( true );
  294.  
  295.                            if( isset( $desc->Key ) && $desc->Key == 'PRI' )
  296.                                   $Column->setPrimaryKey( true );
  297.  
  298.                            if( isset( $desc->Extra ) && $desc->Extra == 'auto_increment' )
  299.                                   $Column->setAutoIncrement( true );
  300.  
  301.                                foreach( $foreignKeys as $fkey ) {
  302.     
  303.                                     if( $fkey->table == $Table->getName() &&
  304.                                         $fkey->column == $Column->getName() ) {
  305.     
  306.                                             $ForeignKey = new ForeignKey( null, $fkey->table, $fkey->column );
  307.                                             $ForeignKey->setName( $fkey->constraint );
  308.                                             $ForeignKey->setType( 'one-to-many' );
  309.                                             $ForeignKey->setReferencedTable( $fkey->referenced_table );
  310.                                             $ForeignKey->setReferencedColumn( $fkey->referenced_column );
  311.                                             $ForeignKey->setReferencedController( ucfirst( $fkey->referenced_table ) . 'Controller' );
  312.                                             $ForeignKey->setOnDelete( $fkey->delete_rule );
  313.                                             $ForeignKey->setOnUpdate( $fkey->update_rule );
  314.     
  315.                                             $Column->setForeignKey( $ForeignKey );
  316.                                         }
  317.                                }
  318.                                   
  319.                                  $Table->addColumn( $Column );
  320.                            }
  321.  
  322.                            $Database->addTable( $Table );       
  323.                 }
  324.  
  325.                 return $Database;
  326.       }
