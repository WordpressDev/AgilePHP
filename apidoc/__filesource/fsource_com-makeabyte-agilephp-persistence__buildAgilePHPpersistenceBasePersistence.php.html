  398.                $sql = 'INSERT INTO ' . $table->getName() . '( ';

  401.                for( $i=0; $i<count( $columns ); $i++ ) {
  402.  
  403.                        if( $columns[$i]->isAutoIncrement() )
  404.                            continue;
  405.  
  406.                        $sql .= $columns[$i]->getName();
  407.  
  408.                        if( ($i + 1) < count( $columns ) )
  409.                            $sql .= ', ';
  410.                }
  411.                $sql .= ' ) VALUES ( ';
  412.                for( $i=0; $i<count( $columns ); $i++ ) {
  413.  
  414.                        if( $columns[$i]->isAutoIncrement() )
  415.                                continue;
  416.  
  417.                        $sql .= '?';
  418.                        $method = $this->toAccessor( $columns[$i]->getModelPropertyName() );
  419.  
  420.                        if( $columns[$i]->isForeignKey() ) {
  421.  
  422.                            // php namespace support - extract the class name from the fully qualified class path
  423.                                $foreignModelPieces = explode( '\\', $columns[$i]->getForeignKey()->getReferencedTableInstance()->getModel() );
  424.                                $foreignModelName = array_pop( $foreignModelPieces );
  425.  
  426.                                // Create accessor names for both the foreign model instance and the foreign model's instance accessor
  427.                         $instanceAccessor = $this->toAccessor( $foreignModelName );
  428.                            $accessor = $this->toAccessor( $columns[$i]->getForeignKey()->getReferencedColumnInstance()->getModelPropertyName() );
  429.  
  430.                            if( is_object( $model->$instanceAccessor() ) ) {
  431.  
  432.                                // Get foreign key value from the referenced field/instance accessor
  433.                                if( $model->$instanceAccessor()->$accessor() != null ) {
  434.                                    array_push( $values, $model->$instanceAccessor()->$accessor() );
  435.                                }
  436.                                else {
  437.                                    // Persist the referenced model instance and use its new id as the foreign key value
  438.                                    $this->persist( $model->$instanceAccessor() );
  439.                                    array_push( $values, $this->pdo->lastInsertId() );
  440.                                }
  441.                            }
  442.                            else {
  443.                                // @todo Extract using the foreign key property (is this bad or should we stick to objects only?)
  444.                                array_push( $values, ($model->$method() == '') ? NULL : $model->$method() );
  445.                            }
  446.                        }
  447.                        else // No foreign key
  448.                            array_push( $values, (($model->$method() == '') ? NULL : $model->$method()) );
  449.  
  450.                        if( ($i + 1) < count( $columns ) )
  451.                            $sql .= ', ';
  452.                }
  453.                $sql .= ' );';
  454.  
  455.                   $this->prepare( $sql );
  456.                  return $this->execute( $values );
  457.         }

  459.         /**
  460.          * Merges/updates a persisted domain model object
  461.          * 
  462.          * @param $model The model object to merge/update
  463.          * @return PDOStatement 
  464.          * @throws AgilePHP_PersistenceException
  465.          */
  466.         public function merge( $model ) {
  467.  
  468.                $this->model = $model;
  469.                $table = $this->getTableByModel( $model );
  470.  
  471.                Logger::debug( 'BasePersistence::merge Performing merge on model \'' . $table->getModel() . '\'.' );
  472.  
  473.                $this->model = $model;
  474.                  $values = array();
  475.                  $cols = array();
  476.                $this->validate( $table );
  477.  
  478.                $sql = 'UPDATE ' . $table->getName() . ' SET ';
  479.  
  480.                  $columns = $table->getColumns();
  481.                  $naCount = 0;
  482.                for( $i=0; $i<count( $columns ); $i++ ) {
  483.  
  484.                        if( $columns[$i]->isPrimaryKey() || $columns[$i]->isAutoIncrement() )
  485.                            continue;
  486.  
  487.                        $accessor = $this->toAccessor( $columns[$i]->getModelPropertyName() );
  488.  
  489.                        // Extract foreign key value from the referenced column
  490.                        if( $columns[$i]->isForeignKey() ) {
  491.  
  492.                            // php namespace support - extract the class name from the fully qualified class path
  493.                                $foreignModelPieces = explode( '\\', $columns[$i]->getForeignKey()->getReferencedTableInstance()->getModel() );
  494.                                $foreignModelName = array_pop( $foreignModelPieces );
  495.  
  496.                                // Create accessor name for the foreign model instance
  497.                         $instanceAccessor = $this->toAccessor( $foreignModelName );
  498.  
  499.                            if( is_object( $model->$instanceAccessor() ) ) {
  500.  
  501.                                // Create name for the foreign model's instance accessor
  502.                                $accessor = $this->toAccessor( $columns[$i]->getForeignKey()->getReferencedColumnInstance()->getModelPropertyName() );
  503.  
  504.                                // Get foreign key value from the referenced instance
  505.                                if( $model->$instanceAccessor()->$accessor() != null ) {
  506.  
  507.                                    array_push( $values, $model->$instanceAccessor()->$accessor() );
  508.                                    // @todo Need a way to check if the model is dirty so we dont waste database calls updating data that already exists.
  509.                                    //         Possibly #@Column interceptor and move away from xml altogether? Getting rid of xml will speed up apps with a
  510.                                    //         large database since a large persistence.xml file will have to be parsed with each page hit... 
  511.                                    $this->merge( $model->$instanceAccessor() );
  512.                                }
  513.                                else {
  514.                                    // Persist the referenced model instance, and use its new id as the foreign key value
  515.                                    $this->persist( $model->$instanceAccessor() );
  516.                                    array_push( $values, $this->pdo->lastInsertId() );
  517.                                }
  518.                            }
  519.                            else {
  520.                                // @todo Extract using the foreign key property (is this bad or should we stick to objects only?)
  521.                                array_push( $values, ($model->$accessor() == '') ? NULL : $model->$accessor() );
  522.                            }
  523.                        }
  524.                        else // not a foreign key
  525.                            array_push( $values, $model->$accessor() );
  526.  
  527.                        array_push( $cols, $columns[$i]->getName() );
  528.                }
  529.  
  530.                $sql .= implode( $cols, '=?, ' ) . '=? WHERE ';
  531.  
  532.                $pkeyColumns = $table->getPrimaryKeyColumns();
  533.                for( $i=0; $i<count( $pkeyColumns ); $i++ ) {
  534.  
  535.                       $accessor = $this->toAccessor( $columns[$i]->getModelPropertyName() );
  536.                       $sql .= $columns[$i]->getName() . '=\'' . $model->$accessor() . '\'';
  537.  
  538.                       if( ($i+1) < count( $pkeyColumns ) )
  539.                           $sql .= ' AND ';
  540.                }
  541.  
  542.                $sql .= ';';
  543.  
  544.                $this->prepare( $sql );
  545.                  return $this->execute( $values );
  546.         }

  596.         /**
  597.          * Attempts to locate the specified model by values. Any fields set in the object are used
  598.          * in search criteria. Alternatively, setRestrictions and setOrderBy methods can be used to
  599.          * filter results.
  600.          * 
  601.          * @param $model A domain model object. Any fields which are set in the object are used to filter results.
  602.          * @throws AgilePHP_PersistenceException If any primary keys contain null values or any
  603.          *             errors are encountered executing queries
  604.          */
  605.         public function find( $model ) {
  606.  
  607.                $table = $this->getTableByModel( $model );
  608.                $newModel = $table->getModelInstance();
  609.                $values = array();
  610.  
  611.                Logger::debug( 'BasePersistence::find Performing find on model \'' . $table->getModel() . '\'.' );
  612.  
  613.                  try {
  614.                           if( $this->isEmpty( $model ) ) {
  615.  
  616.                             $sql = 'SELECT ' . (($this->getDistinct() == null) ? '*' : 'DISTINCT ' . $this->getDistinct()) . ' FROM ' . $table->getName();
  617.  
  618.                             $order = $this->getOrderBy();
  619.                             $offset = $this->getOffset();
  620.                             $groupBy = $this->getGroupBy();
  621.  
  622.                              $sql .= ($this->restrictions != null) ? $this->createRestrictSQL() : '';
  623.                           $sql .= ($order != null) ? ' ORDER BY ' . $order['column'] . ' ' . $order['direction'] : '';
  624.                           $sql .= ($groupBy)? ' GROUP BY ' . $this->getGroupBy() : '';
  625.                           $sql .= ($offset && $this->getMaxResults()) ? ' LIMIT ' . $offset . ', ' . $this->getMaxResults() : '';
  626.                           $sql .= (!$offset && $this->getMaxResults()) ? ' LIMIT ' . $this->getMaxResults() : '';
  627.                              $sql .= ';';
  628.  
  629.                                 $this->setDistinct( null );
  630.                              $this->setRestrictions( array() );
  631.                              $this->setRestrictionsLogicOperator( 'AND' );
  632.                              $this->setOrderBy( null, 'ASC' );
  633.                              $this->setGroupBy( null );
  634.                         }
  635.                         else {
  636.                                 $where = '';
  637.  
  638.                                 $columns = $table->getColumns();
  639.                             for( $i=0; $i<count( $columns ); $i++ ) {
  640.  
  641.                                   $accessor = $this->toAccessor( $columns[$i]->getModelPropertyName() );
  642.                                   if( $model->$accessor() == null ) continue;
  643.  
  644.                                   $where .= (count($values) ? ' ' . $this->restrictionsLogic . ' ' : ' ') . $columns[$i]->getName() . ' ' . $this->comparisonLogic . ' ?';
  645.                                    array_push( $values, $model->$accessor() );
  646.                             }
  647.                             $sql = 'SELECT * FROM ' . $table->getName() . ' WHERE' . $where;
  648.                             $sql .= ' LIMIT ' . $this->getMaxResults() . ';';
  649.                         }
  650.  
  651.                      $this->prepare( $sql );
  652.                      $this->PDOStatement->setFetchMode( PDO::FETCH_OBJ );
  653.                      $result = $this->execute( $values );
  654.  
  655.                      if( !count( $result ) ) {
  656.  
  657.                           Logger::debug( 'BasePersistence::find Empty result set for model \'' . $table->getModel() . '\'.' );
  658.                           return null;
  659.                      }
  660.  
  661.                       $index = 0;
  662.                       $models = array();
  663.                      foreach( $result as $stdClass  ) {
  664.  
  665.                                $m = $table->getModelInstance();
  666.                                   foreach( get_object_vars( $stdClass ) as $name => $value ) {
  667.  
  668.                                              if( !$value ) continue;
  669.                                              $modelProperty = $this->getPropertyNameForColumn( $table, $name );
  670.                                              
  671.                                            // Create foreign model instances from foreign values
  672.                                          foreach( $table->getColumns() as $column ) {
  673.  
  674.                                                     if( $column->isForeignKey() && $column->getName() == $name ) {
  675.  
  676.                                                            $foreignModel = $column->getForeignKey()->getReferencedTableInstance()->getModel();
  677.                                                            $foreignInstance = new $foreignModel();
  678.  
  679.                                                            $foreignMutator = $this->toMutator( $column->getForeignKey()->getReferencedColumnInstance()->getModelPropertyName() );
  680.                                                            $foreignInstance->$foreignMutator( $value );
  681.  
  682.                                                            $persisted = $this->find( $foreignInstance );
  683.  
  684.                                                            // php namespace support - remove \ character from fully qualified paths
  685.                                                            $foreignModelPieces = explode( '\\', $foreignModel );
  686.                                                            $foreignClassName = array_pop( $foreignModelPieces );
  687.  
  688.                                                            $instanceMutator = $this->toMutator( $foreignClassName );
  689.                                                            $m->$instanceMutator( $persisted[0] );
  690.                                                     }
  691.                                                     else {
  692.  
  693.                                                            $mutator = $this->toMutator( $modelProperty );
  694.                                                              $m->$mutator( $value );
  695.                                                     }
  696.                                          }
  697.                                   }
  698.  
  699.                                   array_push( $models, $m );
  700.                                   $index++;
  701.                                   if( $index == $this->maxResults )  break;
  702.                      }
  703.  
  704.                      return $models;
  705.                }
  706.                catch( Exception $e ) {
  707.  
  708.                        throw new AgilePHP_PersistenceException( $e->getMessage(), $e->getCode() );
  709.                }
  710.       }
  711.  
  712.       /**
  713.        * Returns the total number of records in the specified model.
  714.        * 
  715.        * @param Object $model The domain object to get the count for.
  716.        * @return Integer The total number of records in the table.
  717.        */
  718.       public function count( $model ) {
  719.  
  720.                $sql = 'SELECT count(*) as count FROM ' . $this->getTableByModel( $model )->getName();
  721.              $sql .= ($this->createRestrictSQL() == null) ? '' : $this->createRestrictSQL();
  722.              $sql .= ';';
  723.  
  724.               $stmt = $this->prepare( $sql );
  725.               $stmt->execute( array( $this->getTableByModel( $model )->getName() ) );
  726.                $stmt->setFetchMode( PDO::FETCH_OBJ );
  727.                $result = $stmt->fetchAll();
  728.  
  729.                return ($result == null) ? 0 : $result[0]->count;
  730.       }

  736.        */

  740.       }
  741.  
  742.       /**
  743.        * Sets the SQL 'order by' clause.
  744.        * 
  745.        * @param $column The column name to order the result set by
  746.        *  $param $direction The direction to sort the result set (ASC|DESC).

  748.        */
  749.       public function setOrderBy( $column, $direction ) {

  762.       public function getOrderBy() {

  776.       public function setRestrictions( array $restrictions ) {

  779.       }

  781.       /**
  782. /**
  783.        * Returns the WHERE clause restrictions
  784.        * 
  785.        * @return Array SQL WHERE clause restrictions
  786.        */
  787.       public function getRestrictions() {

  792.       /**
  793.        * Sets the restriction operator (and|or) used in SQL WHERE clause.
  794.        * 
  795.        * @param $operator The logical operator 'and'/'or' to be used in SQL WHERE clause. Default is 'AND'.
  796.        * @return void 
  797.        */
  798.       public function setRestrictionsLogicOperator( $operator ) {
  799.  
  800.                 if( strtolower( $operator ) !== 'and' && strtolower( $operator ) !== 'or' )
  801.                   throw new AgilePHP_PersistenceException( 'Restrictions logic operator must be either \'and\' or \'or\'. Found \'' . $operator . '\'.' );
  802.  
  803.               $this->restrictionsLogic = $operator;
  804.       }
  805.  
  806.       /**
  807.        * Sets the comparison operator (<|>|=|LIKE) used in SQL WHERE clause.
  808.        * 
  809.        * @param $operator The logical comparison operator used is SQL where clauses. Default is '='.
  810.        * @return void 
  811.        */
  812.       public function setComparisonLogicOperator( $operator ) {
  813.  
  814.                if( strtolower( $operator ) != 'like' && $operator !== '<' && $operator !== '>' && $operator !== '=' )
  815.                   throw new AgilePHP_PersistenceException( 'Comparison logic operator must be \'>\', \'<\', \'=\', or \'LIKE\'. Found \'' . $operator . '\'.' );
  816.  
  817.               $this->comparisonLogic = $operator;
  818.       }
  819.  
  820.        /**
  821.        * Returns an SQL formatted string containing a WHERE clause built from setRestrictions and setRestrictionsLogicOperator.
  822.        * 
  823.        * @return The formatted SQL string
  824.        */
  825.       public function createRestrictSQL() {
  826.  
  827.               $restricts = null;
  828.              if( count( $this->restrictions ) ) {
  829.  
  830.                    $restricts = ' WHERE ';
  831.                  $index = 0;
  832.                  foreach( $this->restrictions as $key => $val ) {
  833.  
  834.                              $index++;
  835.                              $restricts .= $key . ' ' . $this->comparisonLogic . ' \'' . $val . '\'';
  836.  
  837.                              if( $index < count( $this->restrictions ) )
  838.                                  $restricts .= ' ' . $this->restrictionsLogic . ' ';
  839.                  }
  840.              }
  841.  
  842.              return $restricts;
  843.          }
