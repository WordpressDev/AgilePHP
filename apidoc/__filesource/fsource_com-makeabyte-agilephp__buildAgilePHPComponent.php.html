  172.        * Dispatches a request to the specified controller / action / parameters

  174.        *

  177.        * @param array $parameters An array of parameters to pass into the specified controller / action combo

  181.  
  182.                   $controllerCls = get_class( $controller );
  183.                   $namespace = explode( '\\', $controllerCls );
  184.                   $controllerName = array_pop( $namespace );
  185.  
  186.                   $component = get_class( $this );
  187.  
  188.                     $f = AgilePHP::getFramework()->getWebRoot() . DIRECTORY_SEPARATOR . 'components' . DIRECTORY_SEPARATOR . $component . DIRECTORY_SEPARATOR . 'control';
  189.                     $it = new RecursiveDirectoryIterator( $f );
  190.                   foreach( new RecursiveIteratorIterator( $it ) as $file ) {
  191.  
  192.                                if( substr( $file, -1 ) != '.' && substr( $file, -2 ) != '..' ) {
  193.  
  194.                                       $pieces = explode( DIRECTORY_SEPARATOR, $file );
  195.                                      $item = array_pop( $pieces ); 
  196.  
  197.                                      if( $item == $controllerName . '.php' ) {
  198.  
  199.                                         __autoload( $controllerName );
  200.                                      if( !$parameters ) return $controller->$method();
  201.                                     return call_user_func_array( array( $controller, $method ), $parameters );
  202.                                  }
  203.                             }
  204.                   }
  205.  
  206.                     throw new AgilePHP_Exception( 'The requested component controller \'' . $controllerName . '\' could not be found.' );
