  231.

  242.  
  243.                                if( $column->isPrimaryKey() )
  244.                                      $xsl .= '<input type="hidden" name="' . $name . '" value="' . $value . '"/>';

  246.

  265.  
  266.                                $html .= '<td>Password</td>

  276.

  278.                            else if( $column->isAutoIncrement() ) {

  281.                           $html .= ($this->getMode() == 'merge') ?

  342.       /**
  343.        * Returns an HTML select drop-down which contains a list
  344.        * of values for foreign key columns used in one-to-many
  345.        * relationships.
  346.        * 
  347.        * @return String The populated HTML <select> element
  348.        */
  349.       public function getForeignKeySelection( ForeignKey $foreignKey ) {
  350.  
  351.                $pm = PersistenceManager::getInstance();
  352.  
  353.                $selectedColumn = $foreignKey->getSelectedColumnInstance();
  354.                $selectedProperty = $foreignKey->getSelectedColumnInstance()->getModelPropertyName();
  355.  
  356.                 // Create an instance of the foreign model
  357.                 $foreignInstance = $foreignKey->getReferencedTableInstance()->getModelInstance();
  358.  
  359.                 // Find all foreign models by distinct parent column name
  360.                $foreignModels = $pm->find( $foreignInstance );
  361.  
  362.                // The actual foreign key property name in the child table
  363.                $property = $foreignKey->getColumnInstance()->getModelPropertyName();
  364.  
  365.                $html = '<select name="' . $property . '">
  366.                            <option value="NULL">Choose...</option>';
  367.  
  368.                // Create each of the option values. If the foreign key value matches the
  369.                // selected foreign model property value, the item is shown as the default.
  370.                foreach( $foreignModels as $fModel ) {
  371.  
  372.                          $fAccessor = $pm->toAccessor( $selectedProperty );
  373.  
  374.                          // php namespace support
  375.                              $namespace = explode( '\\', $foreignKey->getReferencedTableInstance()->getModel() );
  376.                              $model = $namespace[count($namespace)-1];
  377.  
  378.                          $fkInstanceAccessor = $pm->toAccessor( $model );
  379.  
  380.                          if( is_object( $this->getModel()->$fkInstanceAccessor() ) && 
  381.                                      $this->getModel()->$fkInstanceAccessor()->$fAccessor() == $fModel->$fAccessor() ) {
  382.  
  383.                                // Assign primary key values delimited by underscore _ character 
  384.                                $primaryKeys = $foreignKey->getReferencedTableInstance()->getPrimaryKeyColumns();
  385.                              $pkeys = '';
  386.                              for( $i=0; $i<count( $primaryKeys ); $i++ ) {
  387.  
  388.                                     $idAccessor = 'get' . ucfirst( $primaryKeys[$i]->getModelPropertyName() );
  389.  
  390.                                     $pkeys .= $this->getModel()->$fkInstanceAccessor()->$idAccessor();
  391.                                     $pkeys .= (((count($primaryKeys)-1) == $i)) ? '' : '_';
  392.                              }
  393.                                              
  394.                                $html .= '<option value="' . $pkeys . '" selected="yes">' . $fModel->$fAccessor() . '</option>';
  395.                          }
  396.                          else {
  397.  
  398.                              // Assign primary key values delimited by underscore _ character 
  399.                                $primaryKeys = $foreignKey->getReferencedTableInstance()->getPrimaryKeyColumns();
  400.                              $pkeys = '';
  401.                              for( $i=0; $i<count( $primaryKeys ); $i++ ) {
  402.  
  403.                                     $idAccessor = 'get' . ucfirst( $primaryKeys[$i]->getModelPropertyName() );
  404.  
  405.                                     $pkeys .= $fModel->$idAccessor();
  406.                                     $pkeys .= (((count($primaryKeys)-1) == $i)) ? '' : '_';
  407.                              }
  408.  
  409.                                $html .= '<option value="' . $pkeys . '">' . $fModel->$fAccessor() . '</option>';
  410.                          }
  411.                }
  412.  
  413.                $html .= '</select>';
  414.  
  415.                return $html;
  416.       }

  418.       /**

  422.        *                             primary keys must be separated by an underscore _ character. Default is null
