  331.                // Make sure email exists!
  332.                $pm = new PersistenceManager();
  333.                $table = $table = $pm->getTableByModelName( $this->getModelName() );
  334.                $emailColumn = $table->getColumnNameByProperty( 'email' );
  335.                $pm->prepare( 'SELECT ' . $emailColumn . ' FROM ' . $table->getName() . ' WHERE ' . $emailColumn . '=? AND username=?;' );
  336.                $params = array( $this->getEmail(),
  337.                                  $this->getUsername() );
  338.                if( !$pm->execute( $params )->fetch() )
  339.                     throw new AgilePHP_Exception( 'The information provided does not match our records.' );
  340.  
  341.                // Send email
  342.                $subject = AgilePHP::getFramework()->getAppName() . ' :: Forgotten Password';
  343.                $body = 'Click on the following link to reset your password: ' . $this->getPasswordResetUrl() . '/' . $token . '/' . $this->session->getSessionId();

  345.                if( !mail( $this->getModel()->getEmail(), $subject, $body, $this->getMailHeaders() ) )

  430.  
  431.                if( !$this->getModel() ) throw new AgilePHP_Exception( 'Identity::login Valid user domain model required' );
  432.  
  433.                Logger::debug( 'Identity::login Authenticating username \'' . $username . '\' with password \'' . $password . '\'.' );
  434.                
  435.                $this->getModel()->setUsername( $username );
  436.  
  437.                if( !$this->getModel() ) return false;
  438.  
  439.                $crypto = new Crypto();
  440.                $hashed = $crypto->getDigest( $password );
  441.  
  442.              if( !preg_match( '/' . $hashed . '/', $this->getPassword() ) )
  443.                  return false;
  444.  
  445.                if( $this->getModel()->getRoles() ) {
  446.  
  447.                     foreach( $this->getModel()->getRoles() as $Role ) {
  448.     
  449.                              if( $this->getModel()->getRoleId() == $Role->getName() ) {
  450.  
  451.                                    $this->setRole( $Role );
  452.                                    break;
  453.                              }
  454.                    }
  455.                }
  456.  
  457.                if( !$this->getModel()->getEnabled() )
  458.                     throw new AgilePHP_AccessDeniedException( 'Your account has been disabled.' );
  459.  
  460.                // The session needs to be persisted first to avoid primary key constraint violation
  461.                  $this->session->set( 'IDENTITY_LOGGEDIN', true );
  462.                $this->session->set( 'IDENTITY_USERNAME', $this->getUsername() );
  463.  
  464.                if( !$this->session->isPersisted() )
