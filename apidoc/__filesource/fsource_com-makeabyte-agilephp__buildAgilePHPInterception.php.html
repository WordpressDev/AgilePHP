  101.       /**
  102.        * Creates a new intercepted target instance. The target is created by modifying
  103.        * the source code of the class being intercepted to *classname*_Intercepted.
  104.        * 
  105.        * @return Object The new intercepted target instance
  106.        */
  107.       public function createInterceptedTarget() {
  108.  
  109.                // php namespace support
  110.              $namespace = explode( '\\', $this->class );
  111.              $className = $namespace[count($namespace)-1];
  112.               $namespace = implode( '\\', $namespace );
  113.                $class = str_replace( '\\', '::', $this->class );
  114.                
  115.                if( class_exists( $class, false ) ) return;
  116.  
  117.                $code = $this->getSourceCode( $this->class );
  118.                $code = preg_replace( '/class\s' . $className . '\s/', 'class ' . $className . '_Intercepted ', $code );
  119.  
  120.              $code = $this->clean( $code );
  121.                if( eval( $code ) === false )
  122.                     throw new AgilePHP_InterceptionException( 'Failed to create intercepted target' );
  123.       }

  125.       /**
  126.        * Creates and loads a dynamic proxy class which performs interceptions
  127.        * on the class created by Interception::createInterceptedTarget().
  128.        * 
  129.        * @return void 
  130.        */
  131.       public function createInterceptorProxy() {
  132.  
  133.                // php namespace support
  134.              $namespace = explode( '\\', $this->class );
  135.              $className = $namespace[count($namespace)-1];
  136.              array_pop( $namespace );
  137.              $namespace = implode( '\\', $namespace );
  138.                $class = str_replace( '\\', '::', $this->class );
  139.  
  140.                if( class_exists( $className, false ) ) return;
  141.  
  142.                $code = ($namespace) ? 'namespace ' . $namespace . ';' : ''; 
  143.                $code .= $this->getSourceCode( 'InterceptorProxy' );
  144.                $code = preg_replace( '/InterceptorProxy/', $className, $code );
  145.  
  146.                $stubs = $this->getMethodStubs();
  147.                $proxyMethods = array( 'getInstance', 'getInterceptedInstance',
  148.                                           '__get', '__set', '__isset', '__unset', '__call' );
  149.  
  150.                $constructor = null;
  151.  
  152.                // Create method stubs in the proxy which match those in the intercepted class
  153.                for( $i=0; $i<count( $stubs['signatures'] ); $i++ ) {
  154.  
  155.                        if( $stubs['methods'][$i] == '__construct' ) {
  156.  
  157.                            $constructor = $stubs['signatures'][$i];
  158.                            continue;
  159.                        }
  160.                        else if( in_array( $stubs['methods'][$i], $proxyMethods ) ) continue;
  161.  
  162.                        $call = $stubs['signatures'][$i] . ' { return $this->__call( "' . $stubs['methods'][$i] . '", array' . $stubs['params'][$i] . ' ); } ';
  163.                        $code = preg_replace( '/\}\s*\?>/m', "\t" . $call . PHP_EOL . '}' . PHP_EOL . '?>', $code );
  164.                }
  165.  
  166.                // Make sure the proxy constructor matches the intercepted class
  167.                if( $constructor )
  168.                     $code = preg_replace( '/public\sfunction\s__construct.*\)/', $constructor, $code );
  169.  
  170.                $code = $this->clean( $code );
  171.                if( eval( $code ) === false )
  172.                     throw new AgilePHP_InterceptionException( 'Failed to create interceptor proxy for \'' . $this->class . '\'.' );
  173.       }
  174.  
  175.       /**
  176.        * Creates public method stubs in the proxy class that match public methods
  177.        * in the intercepted target class. Without this in place, when using reflection
  178.        * on the intercepted target class name, the reflection results will actually be
  179.        * taking place on the InterceptorProxy class and not return expected results.
  180.        * 
  181.        * @return void 
  182.        */
  183.       private function getMethodStubs() {
  184.  
  185.                 $code = $this->getSourceCode( $this->class );
  186.                 preg_match_all( '/(public\s+function\s+(.*?)(\(.*\)))\s/', $code, $matches );
  187.  
  188.                 if( !isset( $matches[1] ) )
  189.                       return array();
  190.  
  191.                 // Parameter names are gotten from the method signature
  192.                 foreach( $matches[3] as &$param ) {
  193.  
  194.                       // Remove type hinting
  195.                      $param = preg_replace( '/[^\$a-zA-Z0-9][a-zA-Z0-9]+?\s/', ' ', $param );
  196.                      $param = preg_replace( '/=\s*\)/', ')', $param ); // @todo spend countless more hours trying to get everything into one regex
  197.                 }
  198.  
  199.                 $a['signatures'] = $matches[1]; 
  200.                 $a['methods'] = $matches[2];
  201.                 $a['params'] = $matches[3];
  202.  
  203.                 return $a;
  204.       }

  206.       /**
  207.        * Strips PHP open/close tags from source code document so it can be

  221.       /**
  222.        * Returns the PHP file content to be parsed.
  223.        * 
  224.        * @return String PHP code
  225.        * @throws AgilePHP_InterceptionException if the source could not be loaded
  226.        */
  227.       public function getSourceCode( $class ) {
  228.  
  229.                // php namespace support
  230.              $namespace = explode( '\\', $class );
  231.               $class = array_pop( $namespace );
  232.               $namespace = implode( DIRECTORY_SEPARATOR, $namespace ) . DIRECTORY_SEPARATOR;
  233.  
  234.              // Load framework classes
  235.              $path = AgilePHP::getFramework()->getFrameworkRoot() . DIRECTORY_SEPARATOR . $namespace . $class . '.php';
  236.              if( file_exists( $path ) ) {
  237.     
  238.                   return file_get_contents( $path );
  239.              }
  240.              $it = new RecursiveDirectoryIterator( AgilePHP::getFramework()->getFrameworkRoot() );
  241.              foreach( new RecursiveIteratorIterator( $it ) as $file ) {
  242.     
  243.                          if( substr( $file, -1 ) != '.' && substr( $file, -2 ) != '..' ) {
  244.     
  245.                                $array = explode( DIRECTORY_SEPARATOR, $file );
  246.                            if( array_pop( $array ) == $class . '.php' ) {
  247.     
  248.                                 return file_get_contents( $file );
  249.                            }
  250.                       }
  251.              }
  252.     
  253.              // Load web application classes
  254.                $path = AgilePHP::getFramework()->getWebRoot() . DIRECTORY_SEPARATOR . 'control' .
  255.                         DIRECTORY_SEPARATOR . $namespace . $class . '.php';
  256.                if( file_exists( $path ) ) {
  257.     
  258.                  return file_get_contents( $path );
  259.                }
  260.                $path = AgilePHP::getFramework()->getWebRoot() . DIRECTORY_SEPARATOR . 'model' .
  261.                         DIRECTORY_SEPARATOR . $namespace . $class . '.php';
  262.              if( file_exists( $path ) ) {
  263.     
  264.                   return file_get_contents( $path );
  265.              }
  266.              $path = AgilePHP::getFramework()->getWebRoot() . DIRECTORY_SEPARATOR . 'classes' .
  267.                       DIRECTORY_SEPARATOR . $namespace . $class . '.php';
  268.                if( file_exists( $path ) ) {
  269.     
  270.                  return file_get_contents( $path );
  271.                }
  272.              $path = AgilePHP::getFramework()->getWebRoot() . DIRECTORY_SEPARATOR . 'components' .
  273.                       DIRECTORY_SEPARATOR . $namespace . $class . '.php';
  274.              if( file_exists( $path ) ) {
  275.     
  276.                  return file_get_contents( $path );
  277.              }
  278.                // Not found in the usual places - perform deep scan
  279.                $it = new RecursiveDirectoryIterator( AgilePHP::getFramework()->getWebRoot() );
  280.              foreach( new RecursiveIteratorIterator( $it ) as $file ) {
  281.     
  282.                          if( substr( $file, -1 ) != '.' && substr( $file, -2 ) != '..'  &&
  283.                                substr( $file, -4 ) != 'view' ) {
  284.     
  285.                                $pieces = explode( DIRECTORY_SEPARATOR, $file );
  286.                            if( array_pop( $pieces ) == $class . '.php' ) {
  287.     
  288.                                 return file_get_contents( $file );
  289.                            }
  290.                       }
  291.              }
  292.  
  293.              throw new AgilePHP_InterceptionException( 'Failed to load source code for class \'' . $class . '\'.' );
  294.       }
  295. }
