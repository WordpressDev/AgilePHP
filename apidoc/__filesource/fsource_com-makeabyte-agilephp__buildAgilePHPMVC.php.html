  179.        * 
  180.        * NOTE: The URI requirement to communicate with this MVC system is as follows
  181.        *       http://domain.com/ScriptName.php/ControllerName/ActionMethod/arg1/arg2/arg3/etc...
  182.        *

  185.       public function processRequest() {
  186.  
  187.                $path = (isset( $_SERVER['PHP_SELF'] )) ? $_SERVER['PHP_SELF'] : '/';
  188.  
  189.                preg_match( '/^.*?\.php(.*)/si', $path, $matches );
  190.            
  191.                if( count( $matches ) ) {
  192.  
  193.                      $this->parameters = explode( '/', $matches[count($matches)-1] );
  194.                    array_shift( $this->parameters );
  195.  
  196.                    // Assign controller and action
  197.                    $controller = (isset($this->parameters[0]) > 0 && $this->parameters[0] != '') ? $this->parameters[0] : $this->getDefaultController(); 
  198.                    $action = (isset($this->parameters[1])) ? $this->parameters[1] : $this->getDefaultAction();
  199.  
  200.                    // Remove controller and action from mvcPieces
  201.                    array_shift( $this->parameters );
  202.                    array_shift( $this->parameters );
  203.  
  204.                    // Security, Security, Security.... 
  205.                    $controller = addslashes( strip_tags( $controller ) );
  206.                    $action = addslashes( strip_tags( $action ) );
  207.  
  208.                    $this->controller = $controller;
  209.                    $this->action = $action;
  210.                }
  211.  
  212.                $this->controller = isset( $controller ) ? $controller : $this->getDefaultController();
  213.                $this->action = isset( $action ) ? $action : $this->getDefaultAction();
  214.  
  215.              // Make sure controllers are loaded from the web application control directory ONLY.
  216.                if( !in_array( $this->controller, get_declared_classes() ) )
  217.                     $this->loadController( $controller );
  218.  
  219.                $oController = new $this->controller;
  220.                $action = $this->action;
  221.  
  222.                // This try/catch statement hides the exception stack of the inner call. This makes debugging difficult.
  223.                //try {
  224.                        if( isset($this->parameters[0]) ) {
  225.  
  226.                            $request = Scope::getRequestScope();
  227.  
  228.                            if( $this->sanitize )
  229.                                foreach( $this->parameters as $key => $val )
  230.                                          $this->parameters[$key] = $request->sanitize( $val );
  231.  
  232.                           // If this is a class thats been intercepted, check both the inner class and the interceptor for
  233.                           // the presence of the requested method/action.
  234.                           if( method_exists( $this->controller, 'getInterceptedInstance' ) ) {
  235.  
  236.                               if( !method_exists( $oController->getInterceptedInstance(), $action ) )
  237.                                   throw new AgilePHP_Exception( 'The specified action \'' . $action . '\' does not exist.' );
  238.                           }
  239.                           else {
  240.  
  241.                               // This is a standard PHP class that hasnt been intercepted
  242.                               if( !method_exists( $this->controller, $action ) )
  243.                                   throw new AgilePHP_Exception( 'The specified action \'' . $action . '\' does not exist.' );
  244.                           } 
  245.  
  246.                           Logger::debug( 'MVC::processRequest Invoking controller \'' . $this->controller . 
  247.                                            '\', action \'' . $this->action . '\', args \'' . implode( ',', $this->parameters  ) . '\'.' );
  248.  
  249.                            call_user_func_array( array( $oController, $action ), $this->parameters ); 
  250.                        }
  251.                        else {
  252.     
  253.                            Logger::debug( 'MVC::processRequest Invoking controller \'' . $this->controller . 
  254.                                            '\', action \'' . $this->action . '\'.' );
  255.     
  256.                            $oController->$action();

  324.       /**
  325.        * Loads a controller class only if it exists in the application controller directory.

  331.       private function loadController( $controller ) {
  332.  
  333.                 $f = AgilePHP::getFramework()->getWebRoot() . DIRECTORY_SEPARATOR . 'control' .
  334.                         DIRECTORY_SEPARATOR . $controller . '.php';
  335.  
  336.                 if( file_exists( $f ) ) {
  337.  
  338.                       __autoload( $controller );
  339.                       return;
  340.                 }
  341.  
  342.                 // Perform deeper scan of control directory
  343.                 $f = AgilePHP::getFramework()->getWebRoot() . DIRECTORY_SEPARATOR . 'control';
  344.                 $it = new RecursiveDirectoryIterator( $f );
  345.               foreach( new RecursiveIteratorIterator( $it ) as $file ) {
  346.     
  347.                           if( substr( $file, -1 ) != '.' && substr( $file, -2 ) != '..' ) {
  348.     
  349.                                  $pieces = explode( DIRECTORY_SEPARATOR, $file );
  350.                                 $item = array_pop( $pieces ); 
  351.  
  352.                                 if( $item == $controller . '.php' ) {
  353.  
  354.                                    __autoload( $controller );
  355.                                 return;
  356.                             }
  357.                        }
  358.               }
  359.  
  360.                 throw new AgilePHP_Exception( 'The requested controller could not be found.' );
  361.       }
