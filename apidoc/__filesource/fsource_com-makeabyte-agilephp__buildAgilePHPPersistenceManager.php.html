  479.          /**
  480.           * Sets the pagination page number and performs an SQL query to populate the 'resultList'
  481.           * and 'resultCount' properties with their appropriate values for the specified page.
  482.           * 
  483.           * @param $pageNumber The page number
  484.           * @return void 
  485.           */
  486.          public function setPage( $pageNumber ) {
  487.  
  488.                     if( !is_numeric( $pageNumber ) || !$pageNumber )
  489.                         $pageNumber = 1;
  490.  
  491.                     $this->page = $pageNumber;
  492.                    $this->executeCountQuery();
  493.  
  494.                    if( $this->page > $this->pageCount )
  495.                        $pageNumber = $this->pageCount;
  496.  
  497.                    $offset = ($this->getPage() - 1) * $this->getMaxResults();
  498.                     if( $offset < 0 ) $offset = 0;
  499.  
  500.                        $this->dialect->setOffset( $offset );
  501.                    $result = $this->find( $this->getModel() );
  502.  
  503.                    if( !$result ) return false;
  504.  
  505.                    $this->resultList = $result;
  506.                    $this->resultCount = count( $result );
  507.  
  508.                    $this->groupBy = null;
  509.                       $this->dialect->setOrderBy( null, 'ASC' );
  510.                       $this->dialect->setRestrictions( array() );
  511.                       $this->dialect->setRestrictionsLogicOperator( 'AND' );
  512.                       $this->dialect->setComparisonLogicOperator( '=' );
  513.  
  514.                    Log::debug( 'BaseModelController::setPage ' . $this->page );
  515.          }
