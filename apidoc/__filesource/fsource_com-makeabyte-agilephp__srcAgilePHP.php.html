  337.       /**
  338.        * Parses AgilePHP configuration file (agilephp.xml) and initalizes the
  339.        * framework according to the specified configuration.
  340.        *

  342.        * @return void

  344.       private function parseXml( $agilephpDotXml = null ) {
  345.  
  346.                   $agilephp_xml = ($agilephpDotXml) ? $agilephpDotXml : $this->getWebRoot() . DIRECTORY_SEPARATOR . 'agilephp.xml';
  347.  
  348.                 if( !file_exists( $agilephp_xml ) )
  349.                       return;
  350.  
  351.                 $xml = simplexml_load_file( $agilephp_xml );
  352.  
  353.                 $dom = new DOMDocument();
  354.                $dom->Load( $agilephp_xml );             
  355.               if( !$dom->validate() ) {
  356.  
  357.                    throw new AgilePHP_Exception( "agilephp.xml Document Object Model validation failed. You can validate your configurations with the DTD at AgilePHP/core/agilephp.dtd." );
  358.                    return;
  359.               }
  360.  
  361.                 if( $xml->mvc ) {
  362.  
  363.                       require_once 'MVC.php';
  364.  
  365.                       $controller = (string)$xml->mvc->attributes()->controller;
  366.                       $action = (string)$xml->mvc->attributes()->action;
  367.                       $renderer = (string)$xml->mvc->attributes()->renderer;
  368.                       $sanitize = (string)$xml->mvc->attributes()->sanitize;
  369.  
  370.                       $mvc = MVC::getInstance();
  371.                       $mvc->setconfig( $controller, $action, $renderer, $sanitize );
  372.                   }
  373.  
  374.                 if( $xml->logger ) {
  375.  
  376.                       require_once 'Logger.php';
  377.                       $level = $xml->logger->attributes()->level;
  378.                       Logger::setLevel( $level );
  379.                         if( $level == 'debug' ) $this->setDebugMode( true );
  380.                 }
  381.       }
