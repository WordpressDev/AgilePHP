  175.           /**
  176.            * Invokes a non-persistent/stateful instance of the requested class/method
  177.            * passing in arguments if any were defined. This non-stateful approach is
  178.            * how most RPC web services work.
  179.            * 
  180.               * @return mixed Returns the result of the invocation
  181.               * @throws AgilePHP_RemotingException
  182.            */
  183.           public function invoke() {

  185.                  $request = Scope::getInstance()->getRequestScope();
  186.  
  187.                  $stateful = $request->getSanitized( 'stateful' );
  188.                  if( $stateful ) $this->invokeStateful();

  190.                  $class = $request->getSanitized( 'class' );
  191.                  $method = $request->getSanitized( 'method' );
  192.                  $constructorArgs = $this->decode( $request->getSanitized( 'constructorArgs' ) );
  193.                  $args = $this->decode( $request->getSanitized( 'parameters' ) );
  194.  
  195.                    Log::debug( 'Remoting::invoke Invoking class \'' . $class . '\', method \'' . $method .
  196.                            '\', constructorArgs \'' . print_r( $constructorArgs, true ) . '\', args \'' . print_r( $args, true ) . '\'.' );
  197.  
  198.                    try {
  199.                          $clazz = new ReflectionClass( $class );
  200.                          $instance = $constructorArgs ? $clazz->newInstanceArgs( (array)$constructorArgs ) : $clazz->newInstance();
  201.                          $m = $clazz->getMethod( $method );
  202.  
  203.                          $this->getRenderer()->render( $args ? $m->invokeArgs( $instance, (array)$args ) : $m->invoke( $instance ) );

  207.                            throw new AgilePHP_RemotingException( $e->getMessage(), $e->getCode() );
