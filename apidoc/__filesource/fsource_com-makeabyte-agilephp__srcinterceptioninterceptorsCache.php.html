  22. /**
  23.  * Allows caching dynamic data that doesn't require real-time rendering.
  24.  * 
  25.  * @author Jeremy Hahn
  26.  * @copyright Make A Byte, inc
  27.  * @package com.makeabyte.agilephp.interception.interceptors
  28.  *  <code>
  29.  *  class MyClass {
  30.  * 
  31.  *  #@Cache
  32.  *  public function index() {
  33.  * 
  34.  *            // A call that might perform some process intensive operation
  35.  *            // or a database call thats only needed once (like building
  36.  *            // a CMS that looks up form input types to build a contact page, etc,
  37.  *            // and once the site is live they dont change)
  38.  *  }
  39.  * 
  40.  *  #@Cache( minutes = 5 )
  41.  *  public function index() {
  42.  * 
  43.  *            // A call that might perform some processing intensive action
  44.  *            // or a database call thats not really needed for every page request
  45.  *  }
  46.  *  }
  47.  *  </code>
  48.  */
  49. #@Interceptor
  50. class Cache {
  51.  
  52.       /**
  53.        * Cache annotation argument containing the number of minutes to cache the intercepted content.
  54.        * Defaults to 0 (never expires).
  55.        *  
  56.        * @var String The number of minutes to store the cached content
  57.        */
  58.       public $minutes = 0;
  59.  
  60.       // Cache file
  61.       private $file;
  62.  
  63.       /**
  64.        * Handles serving up cached content vs real time content.
  65.        * 
  66.        * @param InvocationContext $ic The invocation context instance responsible for the interception
  67.        * @return void 
  68.        */
  69.       #@AroundInvoke
  70.       public function process( InvocationContext $ic ) {
  71.  
  72.                // Build directory and create it if it doesn't exist
  73.                $dir = '.cache';
  74.                if( !file_exists( $dir ) ) mkdir( $dir );
  75.  
  76.                // Build file path
  77.                $name = get_class( $ic->getTarget() );
  78.                $this->file = $dir . DIRECTORY_SEPARATOR . $name. '_' . $ic->getMethod();
  79.  
  80.                // The requested content needs to be served real-time and cached
  81.                if( !file_exists( $this->file ) )
  82.                     return $this->serveAndCache( $ic );
  83.  
  84.                if( !$this->minutes )
  85.  
  86.                     // Cache never expires
  87.                   return $this->serveFromCache();
  88.  
  89.               else if( $this->minutes > 0 ) { // Serve from cache if the file is less than $this->minutes old
  90.  
  91.                        // Convert from seconds to minutes
  92.                        $this->minutes = $this->minutes * 60;
  93.  
  94.                        if( time() - $this->minutes < filemtime( $this->file ) )
  95.                            return $this->serveFromCache();
  96.                }
  97.  
  98.                $this->serveAndCache( $ic );
  99.       }
  100.  
  101.       /**
  102.        * Renders a real-time request and caches the output.
  103.        * 
  104.        * @return void 
  105.        */
  106.       private function serveAndCache( InvocationContext $ic ) {
  107.  
  108.               ob_start();
  109.  
  110.               $clsName = get_class( $ic->getTarget() );
  111.               $o = new $clsName();
  112.  
  113.               $class = new ReflectionClass( $o );
  114.               $m = $class->getMethod( $ic->getMethod() );
  115.               $ic->getParameters() ? $m->invokeArgs( $o, $ic->getParameters() ) : $m->invoke( $o );
  116.  
  117.                  $h = fopen( $this->file, 'w' );
  118.               fwrite( $h, ob_get_contents() );
  119.               fclose( $h );
  120.  
  121.                  ob_end_flush();
  122.  
  123.                  Logger::debug( '#@Cache::serveAndCache Cached ' . $this->file );
  124.       }
  125.  
  126.       /**
  127.        * Serves up cached content with a prefixed HTML comment indicating when the file was cached.
  128.        * 
  129.        * @return void 
  130.        */
  131.       private function serveFromCache() {
  132.  
  133.                echo '<!-- Cached ' . date( 'c', filemtime( $this->file ) ) . "-->\n";
  134.                include $this->file;
  135.       }
  136. }
