  80.                $this->query( 'CREATE DATABASE ' . $this->database->getName() . ';' );
  81.                
  82.                // Close the connection to the server and bind to the new database.
  83.                $this->close();
  84.                $params = array( 'Database' => $this->database->getName(), 'UID' => $this->database->getUsername(), 'PWD' => $this->database->getPassword() );
  85.                if( !$this->conn = sqlsrv_connect( $this->database->getHostname(), $params ) )
  86.                     throw new AgilePHP_PersistenceException( print_r( sqlsrv_errors(), true ) );
  87.  
  88.              $constraintFails = array();
  89.  
  90.                foreach( $this->database->getTables() as $table ) {
  91.  
  92.                                   $sql = $this->toCreateTableSQL( $table );
  93.                                   if( !$this->query( $sql ) ) {
  94.  
  95.                                           $error = sqlsrv_errors();
  96.  
  97.                                           // This saves the create operation from blowing up if persistence.xml defines a table
  98.                                           // that references a table further down the persistence.xml file that has not been
  99.                                           // created yet. Is there a cleaner way - like disabling constraint checks?
  100.                                           if( stristr( $error[0]['message'], 'references invalid table' ) ) {
  101.  
  102.                                                   array_push( $constraintFails, $sql );
  103.                                                   continue;
  104.                                           }
  105.  
  106.                                           throw new AgilePHP_PersistenceException( print_r( sqlsrv_errors(), true ) );
  107.                                   }
  108.                }
  109.  
  110.                // Constraint hack continued
  111.                if( count( $constraintFails ) )
  112.                     foreach( $constraintFails as $sql )
  113.                             if( !$this->query( $sql ) )
  114.                                 throw new AgilePHP_PersistenceException( print_r( sqlsrv_errors(), true ) );          
  115.       }

  136.        * Returns SQL CREATE TABLE statement for the specified table
  137.        *
  138.        * @param Table $table The table to create the SQL code for
  139.        * @return string The SQL CREATE TABLE statement
  140.        */
  141.       private function toCreateTableSQL( Table $table ) {
  142.  
  143.               $sql = 'CREATE TABLE ' . $table->getName() . ' ( ';
  144.  
  145.               foreach( $table->getColumns() as $column ) {
  146.  
  147.                        $sql .= '[' . $column->getName() . '] ' . $column->getType() .
  148.                                  (($column->getLength()) ? '(' . $column->getLength() . ')' : '') .
  149.                                  (($column->isRequired() == true) ? ' NOT NULL' : '') .
  150.                                  (($column->isAutoIncrement() === true) ? ' IDENTITY(1,1)' : '') .
  151.                                  (($column->getDefault() && $column->getType() != 'datetime' &&
  152.                                      $column->getType() != 'timestamp' && !$column->isAutoIncrement() &&
  153.                                      !$column->isPrimaryKey()) ? ' DEFAULT ' . $column->getDefault() : '') .
  154.                                  ((!$column->getDefault() && !$column->isRequired() && !$column->isAutoIncrement() &&
  155.                                       !$column->isPrimaryKey()) ? ' DEFAULT NULL' : '') . ', ';
  156.               }
  157.  
  158.               $pkeyColumns = $table->getPrimaryKeyColumns();
  159.               if( count( $pkeyColumns ) ) {
  160.  
  161.                   $sql .= ' PRIMARY KEY ( ';
  162.                   for( $i=0; $i<count( $pkeyColumns ); $i++ ) {
  163.  
  164.                        $sql .= '[' . $pkeyColumns[$i]->getName() . ']';
  165.  
  166.                        if( ($i+1) < count( $pkeyColumns ) )
  167.                            $sql .= ', ';
  168.                   }
  169.                   $sql .= ' ), ';
  170.               }
  171.  
  172.               if( $table->hasForeignKey() ) {
  173.  
  174.                   $bProcessedKeys = array();
  175.                   $foreignKeyColumns = $table->getForeignKeyColumns();
  176.                   for( $h=0; $h<count( $foreignKeyColumns ); $h++ ) {
  177.  
  178.                         $fk = $foreignKeyColumns[$h]->getForeignKey();
  179.  
  180.                         if( in_array( $fk->getName(), $bProcessedKeys ) )
  181.                             continue;
  182.  
  183.                         $fk->setOnUpdate( str_replace( '_', ' ', $fk->getOnUpdate() ) );
  184.                         $fk->setOnDelete( str_replace( '_', ' ', $fk->getOnDelete() ) );
  185.  
  186.                         // Get foreign keys which are part of the same relationship
  187.                         $relatedKeys = $table->getForeignKeyColumnsByKey( $fk->getName() );
  188.  
  189.                         $sql .= ' CONSTRAINT ' . $fk->getName() . '';
  190.                         $sql .= ' FOREIGN KEY ( ';
  191.  
  192.                         for( $j=0; $j<count( $relatedKeys ); $j++ ) {
  193.  
  194.                              $sql .= $relatedKeys[$j]->getColumnInstance()->getName();
  195.                              if( ($j+1) < count( $relatedKeys ) )
  196.                                  $sql .= ', ';
  197.                         }
  198.  
  199.                         $sql .= ' ) REFERENCES ' . $fk->getReferencedTable() . ' ( ';
  200.  
  201.                         for( $j=0; $j<count( $relatedKeys ); $j++ ) {
  202.  
  203.                              $sql .= '' . $relatedKeys[$j]->getReferencedColumn() . '';
  204.                              if( ($j+1) < count( $relatedKeys ) )
  205.                                  $sql .= ', ';
  206.                         }
  207.  
  208.                         $sql .= ' ) ';
  209.                         $sql .= (($fk->getOnUpdate()) ? ' ON UPDATE ' . $fk->getOnUpdate() : '' );
  210.                         $sql .= (($fk->getOnDelete()) ? ' ON DELETE ' . $fk->getOnDelete() : '' ) . ', ';
  211.  
  212.                         array_push( $bProcessedKeys, $fk->getName() );
  213.                    }
  214.               }
  215.  
  216.               $sql .= ');';
  217.  
  218.               return $sql;
  219.           }

  326.       /**
  327.        * Overrides parent find method to provide MSSQL specific syntax.
  328.        * 
  329.        * @param $model A domain model object. Any fields which are set in the object are used to filter results.
  330.        * @throws AgilePHP_PersistenceException If any primary keys contain null values or any
  331.        *             errors are encountered executing queries
  332.        */
  333.       public function find( $model ) {
  334.  
  335.              $table = $this->getTableByModel( $model );
  336.              $newModel = $table->getModelInstance();
  337.              $values = array();
  338.  
  339.              Log::debug( 'SQLSRVDialect::find Performing find on model \'' . $table->getModel() . '\'.' );
  340.  
  341.                try {
  342.                         $pkeyColumns = $table->getPrimaryKeyColumns();
  343.                          if( $this->isEmpty( $model ) ) {
  344.  
  345.                            $sql = 'SELECT';
  346.  
  347.                            if( $this->getDistinct() != null )                               
  348.                                $sql .= ' DISTINCT ' . $this->getDistinct();
  349.                            else 
  350.                                    $sql .= ($this->getMaxResults() ? ' TOP ' . $this->getMaxResults() . ' *' : '');
  351.  
  352.                            $sql .= ' FROM ' . $table->getName();
  353.  
  354.                            $order = $this->getOrderBy();
  355.                            $offset = $this->getOffset();
  356.                            $groupBy = $this->getGroupBy();
  357.  
  358.                             $sql .= ($this->getRestrictions() != null) ? $this->createRestrictSQL() : '';
  359.                          $sql .= ($order != null) ? ' ORDER BY ' . $order['column'] . ' ' . $order['direction'] : '';
  360.                          $sql .= ($groupBy)? ' GROUP BY ' . $this->getGroupBy() : '';
  361.                             $sql .= ';';
  362.  
  363.                                $this->setDistinct( null );
  364.                             $this->setRestrictions( array() );
  365.                             $this->setRestrictionsLogicOperator( 'AND' );
  366.                             $this->setOrderBy( null, 'ASC' );
  367.                             $this->setGroupBy( null );
  368.                         }
  369.                         else {
  370.                                 $where = '';
  371.  
  372.                                 $columns = $table->getColumns();
  373.                             for( $i=0; $i<count( $columns ); $i++ ) {
  374.  
  375.                                   $accessor = $this->toAccessor( $columns[$i]->getModelPropertyName() );
  376.                                   if( $model->$accessor() == null ) continue;
  377.  
  378.                                   $where .= (count($values) ? ' AND ' : ' ') . $columns[$i]->getName() . '=?';
  379.                                  array_push( $values, $model->$accessor() );
  380.                             }
  381.                             $sql = 'SELECT * FROM ' . $table->getName() . ' WHERE' . $where;
  382.                         }
  383.  
  384.                      $this->prepare( $sql );
  385.                      $this->execute( $values );
  386.  
  387.                      if( !sqlsrv_has_rows( $this->stmt ) ) { 
  388.  
  389.                          Log::debug( 'SQLSRVDialect::find Empty result set for model \'' . $table->getModel() . '\'.' );
  390.                          return array();
  391.                      }
  392.  
  393.                       $index = 0;
  394.                       $models = array();
