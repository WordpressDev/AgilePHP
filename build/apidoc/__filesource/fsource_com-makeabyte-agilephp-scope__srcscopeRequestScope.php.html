  38.        * 
  39.        * $_GET variables are not used in AgilePHP for two reasons:
  40.        * 1) 255 character limit
  41.        * 2) GET requests make XSS attacks easier to execute
  42.        * 
  43.        * @return void 
  44.        */
  45.       private function __construct() {
  46.  
  47.                  foreach( $_POST as $key => $value )
  48.                            $this->store[$key] = $_POST[$key];
  49.  
  50.                 if( isset( $_COOKIE['AGILEPHP_REQUEST_TOKEN'] ) && $_SERVER['REQUEST_METHOD'] == 'POST' ) {
  51.  
  52.                       if( !isset( $this->store['AGILEPHP_REQUEST_TOKEN'] ) || 
  53.                                   $this->store['AGILEPHP_REQUEST_TOKEN'] != $_COOKIE['AGILEPHP_REQUEST_TOKEN'] ) {
  54.  
  55.                             $rt = (!isset( $this->store['AGILEPHP_REQUEST_TOKEN'] ) ) ? null : $this->store['AGILEPHP_REQUEST_TOKEN'];
  56.  
  57.                             Log::debug( 'RequestScope::__construct Found invalid request token \'' . $rt . '\', expected \'' . $_COOKIE['AGILEPHP_REQUEST_TOKEN'] . '\'.' );
  58.                             throw new AgilePHP_Exception( 'Invalid request token \'' . $rt . '\'. Possible Cross-Site Forgery Request (CSFR) attempt.' );
  59.                       }
  60.                       else {
  61.  
  62.                           $this->invalidate();

  107.       /**
  108.        * Returns a sanitized store variable which protects against SQL injection,
  109.        * XSS and XSFR attacks. The variable value is passed first through strip_tags,
  110.        * followed by addslashes and finally htmlspecialchars. Use htmlspecialchars_decode
  111.        * to decode the sanitized value.
  112.        * 
  113.        * @param String $key The key/index of the variable
  114.        * @return String The sanitized value

  116.        */
  117.       public function getSanitized( $key ) {
  118.  
  119.                if( isset( $this->store[$key] ) && !empty( $this->store[$key] ) ) {
  120.  
  121.                     if( is_array( $this->store[$key] ) ) {
  122.  
  123.                          for( $i=0; $i<count($this->store[$key]); $i++ ) {
  124.  
  125.                              $this->store[$key][$i] = htmlspecialchars( addslashes( strip_tags( urldecode( $this->store[$key][$i] ) ) ) );
  126.                          }
  127.                          return $this->store[$key];
  128.                     }
  129.  
  130.                     return htmlspecialchars( addslashes( strip_tags( urldecode( $this->store[$key] ) ) ) );

  134.       }

  155.              $lcase = 'abcdefghijklmnopqrstuvwzyz';
  156.              $ucase = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ';
  157.

  159.              for( $i=0; $i<21; $i++ ) {
  160.  
  161.                     if( rand( 0, 1 ) ) {
  162.  
  163.                           $cRand = rand( 0, 25 );
  164.                            $token .= (rand( 0, 1) ) ? $lcase[$cRand] : $ucase[$cRand];
  165.                     }

  168.                            $nRand = rand( 0, 9 );
  169.                            $token .= $numbers[$nRand];
  170.                     }

  172.  
  173.               setcookie( 'AGILEPHP_REQUEST_TOKEN', $token, time()+3600, '/' ); // 1 hour - AJAX-push friendly
  174.  
  175.                return $token;
