  40. #@Interceptor
  41. class POST {
  42.  
  43.       /**
  44.        * Liason between REST client and service to handle data transformations and providing
  45.        * appropriate "201 Created" HTTP status code header. Missing content-type header result
  46.        * in a 406 Not Acceptable.
  47.        * 
  48.        * @param InvocationContext $ic The intercepted call state
  49.        * @return void 
  50.        */
  51.       #@AroundInvoke
  52.       public function process( InvocationContext $ic ) {
  53.  
  54.                $callee = $ic->getCallee();
  55.              $class = $callee['class'];
  56.  
  57.              // Require client specified content-type header for data transformations
  58.              $contentType = @$_SERVER['CONTENT_TYPE'];
  59.              if( !$contentType ) {
  60.  
  61.                   Log::error( '#@POST::process Missing Content-Type header.' );
  62.                   throw new RestServiceException( 406 );
  63.              }
  64.  
  65.              // Get the negotiated mime type thats used to format the response data 
  66.              $negotiation = MimeUtil::negotiate( $class, $ic->getMethod() );
  67.              $ProduceMime = $negotiation['ProduceMime'];
  68.              $ConsumeMime = $negotiation['ConsumeMime'];
  69.  
  70.                // Read the PUT data
  71.                $data = trim( file_get_contents( 'php://input' ) );
  72.  
  73.                // Transform data if the REST service resource has a #@ConsumeMime annotation
  74.                if( $ConsumeMime )
  75.                     $data = MimeUtil::consumeTransform( $data, $ConsumeMime );
  76.  
  77.                // Add the data to the parameters passed into the intercepted REST resource action method
  78.                $params = $ic->getParameters();
  79.                array_push( $params, $data );
  80.  
  81.                // Execute the REST service resource and store the return value
  82.                $return = call_user_func_array( array( $ic->getTarget(), $ic->getMethod() ), $params ); 
  83.  
  84.                // Format the return value according to the negotiated mime type and exit the application.
  85.                $out = MimeUtil::serverTransform( $return, $ProduceMime );
  86.                header( 'HTTP/1.1 201 Created' );
  87.                die( $out );
  88.       }
