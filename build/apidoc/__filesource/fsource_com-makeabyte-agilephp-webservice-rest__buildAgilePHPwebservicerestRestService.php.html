  22. /**
  23.  * Interceptor responsible for capturing REST resource requests from the MVC
  24.  * component and routing them to the proper service method and resource based
  25.  * on URI composition.
  26.  * 
  27.  * NOTE: It seems that ironically enough, REST does not really fit the MVC paradigm,
  28.  * since MVC requests use /service-or-controller/action-also-known-as-a-verb whereas
  29.  * REST requests use /service-or-controller/nouns-only.
  30.  * The #@RestService acts as a router or front controller to ensure requests to REST
  31.  * resources stay "restful".
  32.  * 
  33.  * @author Jeremy Hahn
  34.  * @copyright Make A Byte, inc
  35.  * @package com.makeabyte.agilephp.webservice.rest
  36.  *  <code>
  37.  *  #@RestService
  38.  *  class MyRestAPI {
  39.  * 
  40.  *           // #@RestService interceptor will route the HTTP request to the proper resource
  41.  *           // defined in the service based on #@Path::resource declarations.
  42.  *  }
  43.  *  </code>
  44.  */
  45. #@Interceptor 
  46. class RestService {
  47.  
  48.       /**
  49.        * Intercepts a REST web service before the MVC executes an
  50.        * action method. This allows introspection of the REST web service
  51.        * as its instantiated so this class can perform routing of the
  52.        * REST request to the proper action method.
  53.        * 
  54.        * @param InvocationContext $ic The call state of the interception
  55.        * @return void 
  56.        * @throws RestException
  57.        */
  58.       #@AroundInvoke
  59.       public function intercept( InvocationContext $ic ) {
  60.  
  61.                // Parse the REST service class name, resource, and parameters from the HTTP URL
  62.                $callee = $ic->getCallee();
  63.                $mvc = $callee['object'];
  64.                $service = $mvc->getController();
  65.                $action = $mvc->getAction();
  66.                $parameters = $mvc->getParameters();
  67.                array_unshift( $parameters, $action );
  68.                $request = '/' . implode( '/', $parameters );
  69.  
  70.                Log::debug( '#@RestService::intercept Routing REST service \'' . $service . '\' resource request \'' . $request . '\'.' );
  71.  
  72.              $annotes = Annotation::getMethodsAsArray( $service );
  73.              foreach( $annotes as $method => $annotations ) {
  74.  
  75.                        // Process default action method first. It will never have any variables to extract
  76.                        if( $request == '/index' ) return $ic->proceed();
  77.  
  78.                        foreach( $annotations as $annote ) {
  79.  
  80.                                 if( $annote instanceof Path ) {
  81.  
  82.                                        // All #@Path annotations should have a resource at this point since the default /index
  83.                                        // request has already been processed. Ignore this method.
  84.                                      if( !$annote->resource ) continue;
  85.  
  86.                                      // No variables to extract simply invoke the requested resource
  87.                                      if( $annote->resource == $request )
  88.                                           call_user_func( array( $service, $method ) );
  89.  
  90.                                      // Create regex based on the #@Path resource definition
  91.                                      $escapedAnnote = preg_replace( '/\//', '\/', $annote->resource );
  92.                                      $regex = '^' . preg_replace( '/{.+?}/', '(.+)', $escapedAnnote ) . '$';
  93.  
  94.                                      // Extract variable names and values
  95.                                      preg_match( '/' . $regex . '/', $annote->resource, $variables ); // Extract variable names from the #@Path::resource definition
  96.                                      preg_match( '/' . $regex . '/', $request, $values );    // Extract values from the request URI
  97.  
  98.                                      if( !isset( $values[1] ) ) continue; // No match
  99.  
  100.                                      // First elements are the text that was matched - remove
  101.                                      array_shift( $variables );
  102.                                      array_shift( $values );
  103.  
  104.                                      // Use the extracted variable names and values to construct #@Path and $request matchers. If
  105.                                      // the $method #@Path resource matches the $request and the extracted values match up with
  106.                                      // EL {variable} braces, then route the call to this method.
  107.                                      $req = preg_replace( '/\//', '\\\\\0', $request );
  108.                                      $resMatcher = "/^$req";
  109.                                      $pathMatcher = "/^$req";
  110.                                      for( $i=0; $i<count( $values ); $i++ ) {
  111.  
  112.                                              $var = preg_replace( '/\{|\}/', '\\\\\0', $variables[$i] ); // escpape { and } chars
  113.                                            $pathMatcher = str_replace( $values[$i], $var, $pathMatcher );
  114.                                            $resMatcher = str_replace( $var, $values[$i], $resMatcher );
  115.                                      }
  116.                                      $pathMatcher .= '$/';
  117.                                      $resMatcher .= '$/';
  118.  
  119.                                      // Execute the REST service method if its #@Path resource matches the current request
  120.                                      if( preg_match( $pathMatcher, $annote->resource ) && preg_match( $resMatcher, $request ) ) {
  121.  
  122.                                             $verb = strtoupper( $_SERVER['REQUEST_METHOD'] );
  123.                                             $hasVerb = false;
  124.                                             foreach( $annotations as $a )
  125.                                                     if( $a instanceof $verb ) $hasVerb = true; 
  126.  
  127.                                             if( !$hasVerb ) continue;
  128.  
  129.                                             $ic->setMethod( $method );
  130.                                          $ic->setParameters( $values );
  131.  
  132.                                          return $ic->proceed();
  133.                                        }
  134.                               }
  135.                      }
  136.              }
  137.  
  138.              Log::debug( '#@RestService::intercept Failed to route \'' . $request . '\' to a \'' . $service . '\' service method.' );
  139.              throw new RestServiceException( 404 );
  140.       }
  141. }
