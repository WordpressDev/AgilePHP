  29. class MimeUtil {
  30.  
  31.       /**
  32.        * Responsible for negotiating a mime type used to send data from the server
  33.        * to the client. This is done by first checking the client preferred mime types
  34.        * and ensuring one of the following mimes is supported:
  35.        * 1) application/xml
  36.        * 2) application/json
  37.        * 3) application/xhtml+xml
  38.        * 
  39.        * The first mime type to match according to the clients preferred list is used.
  40.        * Finally, the REST service action method is introspected in search of a #@ProduceMime
  41.        * annotation. If #@ProduceMime::type is accepted by the client this mime type is returned,
  42.        * otherwise a 406 "Not Acceptable" is returned to the client.
  43.        *
  44.        * @param string $class The service to negotiate with
  45.        * @param string $method The service method to negotiate with
  46.        * @return string The negotiated mime type
  47.        * @static
  48.        */
  49.       public static function negotiate( $class, $method ) {
  50.  
  51.                 $supportedMimes = array( 'application/xml', 'application/json', 'application/x-yaml', 'application/xhtml+xml', '*/*' );
  52.  
  53.               // Get the mime types the client desires from the HTTP_ACCEPT header.
  54.                 // NOTE: This is very primitive/"light-weight" and does not provide full support for RFC 2616 Accept headers,
  55.                 //       however, does support basic content negotiation none the less.
  56.                 $clientMimes = array();
  57.               foreach( explode( ',', $_SERVER['HTTP_ACCEPT'] ) as $mimeType ) {
  58.  
  59.                        $item = explode( ';', $mimeType );
  60.                        $clientMimes[$item[0]] = floatval( array_key_exists( 1, $item ) ? substr( $item[1], 2 ) : 1 );
  61.               }
  62.               arsort( $clientMimes );
  63.  
  64.               // Find the first preferred client mime type which the REST service supports.
  65.               foreach( $clientMimes as $mimeType => $index ) {
  66.  
  67.                        if( in_array( $mimeType, $supportedMimes ) ) {
  68.  
  69.                             $mime = ($mimeType == '*/*') ? 'application/xml' : $mimeType;
  70.                             break;
  71.                        }
  72.               }
  73.  
  74.               // If the client does not accept one of the supported mime types, throw a 406.
  75.               if( !isset( $mime ) ) {
  76.  
  77.                     Log::error( 'MimeUtil::negotiate Client does not accept any of the supported mime types.' );
  78.                     throw new RestServiceException( 406 );
  79.               }
  80.  
  81.               $response = array();
  82.               $response['ConsumeMime'] = @$_SERVER['CONTENT_TYPE'];
  83.  
  84.               // Parse #@ProduceMime and #@ConsumeMime values from the REST service method if present
  85.                 $annotes = Annotation::getMethodsAsArray( $class );
  86.                 foreach( $annotes[$method] as $annotation ) {
  87.  
  88.                           if( $annotation instanceof ProduceMime ) {
  89.  
  90.                                  // If the REST service has a #@ProduceMime::type defined which the client does not accept, send a 406 and exit.
  91.                                  if( !array_key_exists( $annotation->type, $clientMimes ) ) {
  92.  
  93.                                         Log::error( 'MimeUtil::negotiate Client does not accept mime type defined by #@ProduceMime::type' );
  94.                                         throw new RestServiceException( 406 );
  95.                                  }
  96.  
  97.                                  $mime = $annotation->type;
  98.                           }
  99.  
  100.                           if( $annotation instanceof ConsumeMime )
  101.                                  $response['ConsumeMime'] = $annotation->type;
  102.                 }
  103.  
  104.                 $response['ProduceMime'] = $mime;
  105.  
  106.                 return $response;
  107.       }
  108.  
  109.       /**
  110.        * Transforms data being produced by a REST service into a new mime/data type
  111.        * for presentation to the client.
  112.        * 
  113.        * @param mixed $data The data produced by the REST service. THIS DATA MUST BE EITHER AN OBJECT OR AN ARRAY!
  114.        * @param string $mime The mime type which describes the new data formatting.
  115.        *                       (application/xml|application/json|application/x-yaml|application/xhtml+xml)
  116.        */
  117.       public static function serverTransform( $data, $mime ) {
  118.  
  119.                if( !is_object( $data ) && !is_array( $data ) ) {
  120.  
  121.                     Log::debug( 'MimeUtil::serverTransform The specified data must be either an object or array at \'' . $data . '\.' );
  122.                     throw new RestServiceException( 500 );
  123.                }
  124.  
  125.                switch( $mime ) {
  126.  
  127.                    case 'application/xml':
  128.                         header( 'content-type: application/xml' );
  129.                          $renderer = new AJAXRenderer();
  130.                         return $renderer->toXML( $data );
  131.                   break;
  132.  
  133.                    case 'application/json':
  134.                         header( 'content-type: application/json' );
  135.                         $renderer = new AJAXRenderer();
  136.                         return $renderer->toJSON( $data );
  137.                    break;
  138.  
  139.                    case 'application/x-yaml':
  140.                          $renderer = new AJAXRenderer();
  141.                          header( 'content-type: application/x-yaml' );
  142.                        return $renderer->toYAML( $data );
  143.                    break;
  144.  
  145.                    case 'application/xhtml+xml':
  146.                          $renderer = new AJAXRenderer();
  147.                          header( 'content-type: application/xml' );
  148.                          return $renderer->toXML( $data );
  149.                    break;
  150.  
  151.                    default:
  152.                        Log::debug( 'MimeUtil::serverTransform Could not produce unsupported mime type \'' . $mime . '\'.' );
  153.                        throw new RestServiceException( 500 );
  154.                }
  155.       }
  156.  
  157.       /**
  158.        * Transforms the data consumed from the client request into
  159.        * an object which represents the specified mime type. This data
  160.        * is then presented to the requsted REST service resource.
  161.        * 
  162.        * 1) application/xml       = SimpleXMLElement
  163.        * 2) application/json      = JSON unserialized string
  164.        * 3) application/x-yaml    = YAML unserialized string
  165.        * 4) application/xhtml+xml = Data is returned untouched
  166.        * 
  167.        * @param string $data The data being consume from a cilent HTTP request (PUT|POST|DELETE)
  168.        * @param string $mime The mime type which describes the data.
  169.        *                       (application/xml|application/json|application/x-yaml|application/xhtml+xml)
  170.        * @return void 
  171.        */
  172.       public static function consumeTransform( $data, $mime ) {
  173.  
  174.              switch( $mime ) {
  175.  
  176.                  case 'application/xml':
  177.                      if( !$xml = @simplexml_load_string( $data ) ) {
  178.  
  179.                          Log::debug( 'MimeUtil::consumeTransform Malformed data could not be transformed to SimpleXMLElement.' );
  180.                          throw new RestServiceException( 406 );
  181.                      }
  182.                      return $xml;
  183.                  break;
  184.  
  185.                  case 'application/json':
  186.                      return json_decode( $data );
  187.                  break;
  188.  
  189.                  case 'application/x-yaml':
  190.                       return yaml_parse( $data );
  191.                  break;
  192.  
  193.                  case 'application/xhtml+xml':
  194.                       return $data;
  195.                  break;
  196.  
  197.                  default:
  198.                      Log::debug( 'MimeUtil::consumeTransform Could not consume unsupported mime type \'' . $mime . '\'.' );
  199.                      throw new RestServiceException( 406 );
  200.              }
  201.       }
