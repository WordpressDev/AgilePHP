  54.       /**
  55.        * Generates and outputs a WSDL file according to the #@SOAPBinding in the
  56.        * implementing #@WebService.
  57.        * 
  58.        * @param InvocationContext $ic The intercepted call context
  59.        * @return void 
  60.        */
  61.       #@AroundInvoke
  62.       public function generateWSDL( InvocationContext $ic ) {
  63.  
  64.                $callee = $ic->getCallee();
  65.                $this->class = $callee['class']; // The PHP standard class to expose via SOAP
  66.  
  67.                // Supply default web service configuration values
  68.                $this->serviceName = $this->class . 'Service';
  69.               $this->targetNamespace = 'http://' . $_SERVER['HTTP_HOST'] . AgilePHP::getFramework()->getRequestBase() . '/' . MVC::getInstance()->getController();
  70.  
  71.                $annotations = Annotation::getClassAsArray( $this->class );
  72.  
  73.                // Initalize web service configuration from #@WebService annotation if present
  74.                if( count( $annotations ) ) {
  75.  
  76.                     foreach( $annotations as $annotation ) {
  77.  
  78.                             if( $annotation instanceof WebService ) {
  79.  
  80.                                 if( $annotation->serviceName )
  81.                                     $serviceName = $annotation->serviceName;
  82.  
  83.                                 if( $annotation->targetNamespace )
  84.                                     $this->targetNamespace = $annotation->targetNamespace;
  85.                             }
  86.  
  87.                             if( $annotation instanceof SOAPBinding )
  88.                                 $this->SOAPBinding = $annotation;
  89.                     }
  90.                }
  91.  
  92.                // Set default SOAPBinding if not explicitly defined
  93.                if( !$this->SOAPBinding )
  94.                     $this->SOAPBinding = new SOAPBinding();
  95.  
  96.                $clazz = new ReflectionClass( $ic->getTarget() );
  97.                $methods = Annotation::getMethodsAsArray( $this->class );
  98.  
  99.                foreach( $methods as $method => $annotations ) {
  100.  
  101.                          $isWSDLMethod = false;
  102.                          $isWebMethod = false;
  103.                          $parts = '';
  104.  
  105.                          $m = $clazz->getMethod( $method );
  106.  
  107.                        foreach( $annotations as $annote ) {
  108.  
  109.                                if( $annote instanceof WSDL ) {
  110.  
  111.                                    $isWSDLMethod = true;
  112.                                    break;
  113.                                }
  114.  
  115.                                // Include only methods annotated with #@WebMethod annotation
  116.                                $isWebMethod = ($annote instanceof WebMethod || $isWebMethod == $method) ? $method : false;
  117.                                if( !$isWebMethod ) continue;
  118.                        }
  119.  
  120.                        if( $isWSDLMethod ) continue;
  121.  
  122.                        // Document/Literal Wrapped
  123.                          if( $this->SOAPBinding->style == SOAPStyle::DOCUMENT &&
  124.                                        $this->SOAPBinding->use == SOAPStyle::LITERAL ) {
  125.  
  126.                                    $this->createWrappedMethod( $m );
  127.                                   $this->createWrappedMessage( $method );
  128.                                   $this->createPortTypeOperation( $method );
  129.                                   $this->createBindingOperation( $method );
  130.                                   continue;
  131.                        }
  132.                        
  133.                        // RPC/ENCODED/LITERAL
  134.                        foreach( $m->getParameters() as $param ) {
  135.                            
  136.                                $dataType = $this->getParameterTypeFromDocBlock( $m, $param );
  137.                                $ns = $this->getTypeNamespace( $param->name, $param->name );
  138.                                if( preg_match( '/\[\]/', $dataType ) ) {
  139.  
  140.                                    $ns = 'tns';
  141.                                    $dataType = preg_replace( '/\[\]/', 'Array', $dataType );
  142.                                    $this->arrayToComplexType( $dataType, $ns, $param->name );
  143.                                }
  144.                                $parts .= "\t\t<part name=\"" . $param->name . "\" type=\"$ns:" . $dataType . '"/>' . PHP_EOL;
  145.                         }
  146.  
  147.                         $returnType = $this->getReturnTypeFromDocBlock( $m );
  148.                         $ns = $this->getTypeNamespace( $returnType, 'return' );
  149.  
  150.                       $this->createMessage( $method, $parts, $returnType, $ns );
  151.                       $this->createPortTypeOperation( $method );
  152.                       $this->createBindingOperation( $method );
  153.                 }
  154.  
  155.               $xml = $this->assemble();
  156.  
  157.               header( 'content-type: text/xml' );
  158.               header( 'content-length: ' . strlen( $xml ) );
  159.               print( $xml );
  160.       }
  161.  
  162.       /**
  163.        * Adds all of the generated pieces together to form the complete WSDL document.
  164.        * 
  165.        * @return string The WSDL document
  166.        */
  167.       private function assemble() {
  168.  
  169.                 $types = $this->getTypes();
  170.  
  171.                 $xml = "<?xml version=\"1.0\" encoding=\"UTF-8\" ?>" . PHP_EOL;
  172.                 $xml .= "<definitions name=\"" . $this->class . "\" targetNamespace=\"" . $this->targetNamespace . "\"" . PHP_EOL;
  173.                 $xml .= "\txmlns:tns=\"" . $this->targetNamespace . "\" xmlns:soap=\"http://schemas.xmlsoap.org/wsdl/soap/\"" . PHP_EOL;
  174.                 $xml .= "\txmlns:xsd=\"http://www.w3.org/2001/XMLSchema\" xmlns:soap-enc=\"http://schemas.xmlsoap.org/soap/encoding/\"" . PHP_EOL;
  175.                 $xml .= "\txmlns:wsdl=\"http://schemas.xmlsoap.org/wsdl/\" xmlns=\"http://schemas.xmlsoap.org/wsdl/\">" . PHP_EOL . PHP_EOL;
  176.  
  177.                 if( $types ) $xml .= $types;
  178.  
  179.                 $xml .= $this->messages;
  180.                 $xml .= PHP_EOL . "\t<portType name=\"" . $this->class . "PortType\">" . PHP_EOL;
  181.                 $xml .= $this->ptOperations;
  182.                 $xml .= "\t</portType>" . PHP_EOL;
  183.  
  184.                 $xml .= PHP_EOL . "\t<binding name=\"" . $this->class . 'Binding" type="tns:' . $this->class . "PortType\">" . PHP_EOL;
  185.                 $xml .= "\t\t<soap:binding style=\"" . $this->SOAPBinding->style . "\" transport=\"http://schemas.xmlsoap.org/soap/http\"/>" . PHP_EOL;
  186.                 $xml .= $this->bOperations;
  187.                 $xml .= "\t</binding>" . PHP_EOL;
  188.                 $xml .= PHP_EOL . "\t<service name=\"" . $this->serviceName . "\">" . PHP_EOL;
  189.                 $xml .= "\t\t<port name=\"" . $this->class . "Port\" binding=\"tns:" . $this->class . "Binding\">" . PHP_EOL;
  190.                 $xml .= "\t\t\t<soap:address location=\"" . $this->targetNamespace . "\"/>" . PHP_EOL;
  191.                 $xml .= "\t\t</port>" . PHP_EOL;
  192.                 $xml .= "\t</service>" . PHP_EOL;
  193.                 $xml .= '</definitions>';
  194.  
  195.                 return $xml;
  196.       }

  198.       /**
  199.        * Creates a single messages part and appends it to the messages class property.
  200.        * 
  201.        * @param string $method The name of the message part
  202.        * @param string $part The request part
  203.        * @param string $returnType The PHP data type being returned from this message part (extracted from PHP-doc Comment)
  204.        * @return void 
  205.        */
  206.       private function createMessage( $method, $part, $returnType, $ns ) {
  207.           
  208.                 if( stristr( $returnType, '[]' ) )
  209.                       $returnType = preg_replace( '/\[\]/', 'Array', $returnType );
  210.  
  211.                 $messages = "\t<message name=\"" . $method . "Request\">" . PHP_EOL;
  212.                 $messages .= $part;
  213.                 $messages .= "\t</message>" . PHP_EOL;
  214.  
  215.                 $messages .= "\t<message name=\"" . $method . "Response\">" . PHP_EOL;
  216.                 if( $returnType ) $messages .= "\t\t<part name=\"" . $method . "Return\" type=\"$ns:$returnType\" />" . PHP_EOL;
  217.                 $messages .= "\t</message>" . PHP_EOL;
  218.  
  219.                 $this->messages .= $messages;
  220.       }
  221.  
  222.       /**
  223.        * Creates a single message and its part for use with Document/Literal (wrapeed) and appends it to the messages property.
  224.        * 
  225.        * @param string $method The name of the message part
  226.        * @param string $part The request part
  227.        * @param string $returnType The PHP data type being returned from this message part (extracted from PHP-doc Comment)
  228.        * @return void 
  229.        */
  230.       private function createWrappedMessage( $method ) {
  231.  
  232.                 $messages = "\t<message name=\"" . $method . "\">" . PHP_EOL;
  233.                 $messages .= "\t\t<part name=\"parameters\" element=\"tns:" . $method . "\"/>" . PHP_EOL;
  234.                 $messages .= "\t</message>" . PHP_EOL;
  235.  
  236.                 $messages .= "\t<message name=\"" . $method . "Response\">" . PHP_EOL;
  237.                 $messages .= "\t\t<part name=\"parameters\" element=\"tns:" . $method . "Response\"/>" . PHP_EOL;
  238.                 $messages .= "\t</message>" . PHP_EOL;
  239.  
  240.                 $this->messages .= $messages;          
  241.       }
  242.  
  243.       /**
  244.        * Creates a new wsdl binding operation and appends it to the operations class property.
  245.        * 
  246.        * @param string $method The name of the operation to bind
  247.        * @return void 
  248.        */
  249.       private function createBindingOperation( $method ) {
  250.  
  251.                 $operation = "\t\t<wsdl:operation name=\"$method\">" . PHP_EOL;
  252.                 $operation .= "\t\t\t<soap:operation/>" . PHP_EOL;
  253.                 $operation .= "\t\t\t\t<input>" . PHP_EOL;
  254.                 $operation .= "\t\t\t\t\t<soap:body use=\"" . $this->SOAPBinding->use . "\"" . (($this->SOAPBinding->use == SOAPStyle::ENCODED) ? ' encodingStyle="http://schemas.xmlsoap.org/soap/encoding/"' : '') . "/>" . PHP_EOL;
  255.               $operation .= "\t\t\t\t</input>" . PHP_EOL;
  256.               $operation .= "\t\t\t\t<output>" . PHP_EOL;
  257.               $operation .= "\t\t\t\t\t<soap:body use=\"" . $this->SOAPBinding->use . "\"" . (($this->SOAPBinding->use == SOAPStyle::ENCODED) ? ' encodingStyle="http://schemas.xmlsoap.org/soap/encoding/"' : '') . "/>" . PHP_EOL;
  258.                 $operation .= "\t\t\t\t</output>" . PHP_EOL;
  259.                 $operation .= "\t\t</wsdl:operation>" . PHP_EOL;
  260.  
  261.                 $this->bOperations .= $operation;
  262.       }
  263.  
  264.       /**
  265.        * Creates a Port Type operation and appends it to the class property ptOperations.
  266.        * 
  267.        * @param string $method The name of the port type operation
  268.        * @return void 
  269.        */
  270.       private function createPortTypeOperation( $method ) {
  271.  
  272.               $operation = "\t\t<wsdl:operation name=\"$method\">" . PHP_EOL;
  273.               $operation .= "\t\t\t<input message=\"tns:" . $method . "" .
  274.                        (($this->SOAPBinding->style == SOAPStyle::DOCUMENT &&
  275.                             $this->SOAPBinding->use == SOAPStyle::LITERAL) ? '" name="' . $method . '' : 'Request') . "\"/>" . PHP_EOL;
  276.               $operation .= "\t\t\t<output message=\"tns:" . $method . "" .

  279.               $operation .= "\t\t</wsdl:operation>" . PHP_EOL;

  282.       }
  283.  
  284.       /**
  285.        * Creates the wsdl:types node containing each of the complexType's used in the web service
  286.        * 
  287.        * @return String The complexType elements represented in XML
  288.        */
  289.       private function getTypes() {
  290.  
  291.                 if( !$this->types ) return;
  292.  
  293.                 $xml = "\t<wsdl:types>" . PHP_EOL;
  294.                 $xml .= "\t\t<xsd:schema targetNamespace=\"" . $this->targetNamespace . "\">" . PHP_EOL;
  295.               $xml .= $this->types;
  296.               $xml .= "\t\t</xsd:schema>" . PHP_EOL;
  297.                 $xml .= "\t</wsdl:types>" . PHP_EOL;
  298.  
  299.                 return $xml;
  300.       }
  301.  
  302.       /**
  303.        * Creates a new complex type node for the specified class / data type.
  304.        * 
  305.        * @param mixed $class The name of the data type
  306.        * @param string $ns The namespace used for this data type
  307.        * @return void 
  308.        */
  309.       private function objectToComplexType( $class, $ns, $isWrapped = false ) {
  310.  
  311.                 if( in_array( $class, $this->processedComplexObjects ) ) return;
  312.  
  313.                 $eol = PHP_EOL;
  314.  
  315.                 $reflClass = new ReflectionClass( $class );
  316.               $xml = "\t\t\t<xsd:complexType name=\"" . $reflClass->getName() . "\">" . $eol;
  317.               $xml .= "\t\t\t\t<xsd:sequence>" . $eol;
  318.               
  319.               foreach( $reflClass->getProperties() as $property ) {
  320.  
  321.                          $type = $this->getPropertyTypeFromDocBlock( $property );
  322.  
  323.                          if( $this->SOAPBinding->use == SOAPStyle::LITERAL &&    preg_match( '/\[\]/', $type ) ) {
  324.  
  325.                                 $type = preg_replace( '/\[\]/', '', $type );
  326.                                 $ns = $this->getTypeNamespace( $type, $property->name );
  327.                                 $xml .= "\t\t\t\t\t<xsd:element name=\"" . $property->name . "\" minOccurs=\"0\" maxOccurs=\"unbounded\" type=\"" . $ns . ":" . $type . "\"/>" . $eol;
  328.                          }
  329.                          else {
  330.  
  331.                                 $ns = $this->getTypeNamespace( $type, $property->name );
  332.                               $xml .= "\t\t\t\t\t<xsd:element name=\"" . $property->name . "\" type=\"" . $ns . ":" . $type . "\"/>" . $eol;
  333.                          }
  334.               }
  335.  
  336.               $xml .= "\t\t\t\t</xsd:sequence>" . $eol;
  337.               $xml .= "\t\t\t</xsd:complexType>" . $eol;
  338.  
  339.               $this->types .= $xml;
  340.  
  341.               array_push( $this->processedComplexObjects, $class );
  342.       }
  343.  
  344.       /**
  345.        * Creates a Document/Literal wrapped complex type
  346.        * 
  347.        * @return void 
  348.        */
  349.       private function createWrappedMethod( ReflectionMethod $method ) {
  350.  
  351.                 $this->getTypeNamespace( $method->getName(), $method->getName() );
  352.                 $this->getTypeNamespace( $method->getName() . 'Response', $method->getName() );
  353.  
  354.                 $eol = PHP_EOL;
  355.  
  356.                 $xml = "\t\t\t<xsd:element name=\"" . $method->getName() . "\" type=\"tns:" . $method->getName() . "\"/>" . $eol;
  357.                 $xml .= "\t\t\t<xsd:element name=\"" . $method->getName() . "Response\" type=\"tns:" . $method->getName() . "Response\"/>" . $eol;
  358.                 $this->types .= $xml;
  359.  
  360.                 return;
  361.       }
  362.  
  363.       /**
  364.        * Creates a new complex type node for the specified array.
  365.        * 
  366.        * @param string $name The complexType name
  367.        * @param string $ns The namespace to use as a reference for the data type
  368.        * @return void 
  369.        */
  370.       private function arrayToComplexType( $name, $ns, $paramName ) {
  371.  
  372.                 if( in_array( $name, $this->processedComplexArrays ) ) return;
  373.  
  374.                 $eol = PHP_EOL;
  375.  
  376.                 if( $this->SOAPBinding->use == SOAPStyle::ENCODED ) {
  377.  
  378.                       $type = preg_replace( '/Array/', '[]', $name );
  379.                     $name = preg_replace( '/\[\]/', '', $name );
  380.  
  381.                   $xml = "\t\t\t<xsd:complexType name=\"" . $name . "\">" . $eol;
  382.                   $xml .= "\t\t\t\t<xsd:complexContent>" . $eol;
  383.                   $xml .= "\t\t\t\t\t<xsd:restriction base=\"soap-enc:Array\">" . $eol;
  384.                   $xml .= "\t\t\t\t\t\t<xsd:attribute ref=\"soap-enc:arrayType\" wsdl:arrayType=\"" . $ns . ":" . $type . "\"/>" . $eol;
  385.                   $xml .= "\t\t\t\t\t</xsd:restriction>" . $eol;
  386.                   $xml .= "\t\t\t\t</xsd:complexContent>" . $eol;
  387.                   $xml .= "\t\t\t</xsd:complexType>" . $eol;
  388.                 }
  389.  
  390.                 else if( $this->SOAPBinding->use == SOAPStyle::LITERAL ) {
  391.  
  392.                       $type = $name;
  393.  
  394.                       $xsdTypes = array( 'string', 'int', 'float', 'double', 'boolean' );
  395.  
  396.                       preg_match_all( '/Array/', $name, $matches );
  397.                       if( count( $matches[0] ) == 1 ) {
  398.  
  399.                           $rawType = preg_replace( '/Array/', '', $name );
  400.                           if( in_array( $rawType, $xsdTypes ) ) {
  401.  
  402.                                 $type = $rawType;
  403.                                 $ns = 'xsd';
  404.                           }
  405.                           else {
  406.                               
  407.                                 $type = $rawType;
  408.                                 $ns = 'tns';
  409.                           }
  410.                       }
  411.                       else {
  412.                           
  413.                             $type = preg_replace( '/Array/', '', $name, 1 );
  414.                       }
  415.                       
  416.                       $xml = "\t\t\t<xsd:complexType name=\"" . $name . "\">" . $eol;
  417.                   $xml .= "\t\t\t\t<xsd:sequence>" . $eol;
  418.                   $xml .= "\t\t\t\t\t\t<xsd:element minOccurs=\"0\" maxOccurs=\"unbounded\" name=\"$paramName\" type=\"" . $ns . ":" . $type . "\"/>" . $eol;
  419.                   $xml .= "\t\t\t\t</xsd:sequence>" . $eol;
  420.                   $xml .= "\t\t\t</xsd:complexType>" . $eol;
  421.                 }
  422.  
  423.               $this->types .= $xml;
  424.  
  425.               array_push( $this->processedComplexArrays, $name );
  426.       }
  427.  
  428.       /**
  429.        * Gets a property / field level data type from the web service PHP-doc comments block for the specified property
  430.        * 
  431.        * @param ReflectionProperty $property A PHP ReflectionProperty instance representing the property to extract the data type for.
  432.        * @return string The extracted PHP data type
  433.        */
  434.       private function getPropertyTypeFromDocBlock( ReflectionProperty $property ) {
  435.  
  436.                 preg_match( '/@var\\s*(.*?)\\s/i', $property->getDocComment(), $matches );
  437.                 return (isset( $matches[1] )) ? trim( $matches[1] ) : 'anyType';                      
  438.       }
  439.  
  440.       /**
  441.        * Gets a param data type from the web service PHP-doc comments block for the method being inspected.
  442.        * 
  443.        * @param ReflectionMethod $method A PHP ReflectionMethod instance representing the method to extract the parameter data type for.
  444.        * @return string The extracted PHP data type
  445.        */
  446.       private function getParameterTypeFromDocBlock( ReflectionMethod $method, $param ) {
  447.  
  448.                 preg_match( '/@param\\s*(.*?\\[?\\]?)\\s*\$' . $param->name . '/i', $method->getDocComment(), $matches );
  449.                 return (isset( $matches[1] )) ? trim( $matches[1] ) : 'anyType';                      
  450.       }
  451.  
  452.       /**
  453.        * Gets a web servie method return type from the PHP-doc comments block on the method being inspected.
  454.        * 
  455.        * @param ReflectionMethod $method A PHP ReflectionMethod instance representing the method to extract the parameter data type for.
  456.        * @return string The extracted PHP data type
  457.        */
  458.       private function getReturnTypeFromDocBlock( ReflectionMethod $method ) {
  459.  
  460.                 preg_match( '/@return\\s*(.*?)\\s/i', $method->getDocComment(), $matches );
  461.                 return (isset( $matches[1] )) ? trim( $matches[1] ) : null;
  462.       }
  463.  
  464.       /**
  465.        * Returns the WSDL namespace value based on data type.
  466.        * 
  467.        * @param string $type The data type to inspect
  468.        * @return string The namespace value to assign
  469.        */
  470.       private function getTypeNamespace( $type, $paramName ) {
  471.  
  472.                 // This is an array
  473.                 if( preg_match( '/\[\]/', $type ) ) {
  474.  
  475.                     $type = preg_replace( '/\[\]/', 'Array', $type );
  476.                     $ns = 'tns';
  477.                     $this->arrayToComplexType( $type, $ns, $paramName );
  478.                     return $ns;
  479.                 }
  480.  
  481.                 // The type is not an array
  482.                 if( class_exists( $type, false ) ) {
  483.  
  484.                       // The type is an object
  485.                     $ns = 'tns';
  486.                     $this->objectToComplexType( $type, $ns );
  487.  
  488.                     return $ns;
  489.                 }
  490.                 else {
  491.  
  492.                       // The type is primitive
  493.                       return 'xsd';
  494.                 }
  495.       }
